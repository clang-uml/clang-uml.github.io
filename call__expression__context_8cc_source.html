102void call_expression_context::update(clang::FunctionDecl *function)
103{
104 if (!function->isCXXClassMember())
105 reset();
106
107 current_function_decl_ = function;
108
109 // Check if this function is a part of template function declaration,
110 // If no - reset the current_function_template_decl_
111 if ((current_function_template_decl_ != nullptr) &&
112 current_function_template_decl_->getQualifiedNameAsString() !=
113 function->getQualifiedNameAsString()) {
114 current_function_template_decl_ = nullptr;
115 }
116}
117
118void call_expression_context::update(
119 clang::FunctionTemplateDecl *function_template)
120{
121 current_function_template_decl_ = function_template;
122
123 if (!function_template->isCXXClassMember())
124 current_class_decl_ = nullptr;
125
126 current_function_template_decl_ = function_template;
127}

311bool call_expression_context::is_expr_in_current_control_statement_condition(
312 const clang::Stmt *stmt) const
313{
314 if (current_ifstmt() != nullptr) {
315 if (common::is_subexpr_of(current_ifstmt()->getCond(), stmt))
316 return true;
317
318 if (const auto *condition_decl_stmt = current_ifstmt()->getInit();
319 condition_decl_stmt != nullptr) {
320 if (common::is_subexpr_of(condition_decl_stmt, stmt))
321 return true;
322 }
323
324 if (current_elseifstmt() != nullptr) {
325 if (common::is_subexpr_of(current_elseifstmt()->getCond(), stmt))
326 return true;
327 }
328 }
329
330 if (current_conditionaloperator() != nullptr) {
331 if (common::is_subexpr_of(
332 current_conditionaloperator()->getCond(), stmt))
333 return true;
334 }
335
336 if (const auto *loop_stmt = current_loopstmt(); loop_stmt != nullptr) {
337 if (const auto *for_stmt = clang::dyn_cast<clang::ForStmt>(loop_stmt);
338 for_stmt != nullptr) {
339 if (common::is_subexpr_of(for_stmt->getCond(), stmt)) {
340 return true;
341 }
342 if (common::is_subexpr_of(for_stmt->getInit(), stmt)) {
343 return true;
344 }
345 if (common::is_subexpr_of(for_stmt->getInc(), stmt)) {
346 return true;
347 }
348 }
349
350 if (const auto *range_for_stmt =
351 clang::dyn_cast<clang::CXXForRangeStmt>(loop_stmt);
352 range_for_stmt != nullptr) {
353 if (common::is_subexpr_of(range_for_stmt->getRangeInit(), stmt)) {
354 return true;
355 }
356 }
357
358 if (const auto *while_stmt =
359 clang::dyn_cast<clang::WhileStmt>(loop_stmt);
360 while_stmt != nullptr) {
361 if (common::is_subexpr_of(while_stmt->getCond(), stmt)) {
362 return true;
363 }
364 }
365
366 if (const auto *do_stmt = clang::dyn_cast<clang::DoStmt>(loop_stmt);
367 do_stmt != nullptr) {
368 if (common::is_subexpr_of(do_stmt->getCond(), stmt)) {
369 return true;
370 }
371 }
372
373 if (current_conditionaloperator() != nullptr) {
374 if (common::is_subexpr_of(
375 current_conditionaloperator()->getCond(), stmt)) {
376 return true;
377 }
378 }
379 }
380
381 return false;
382}
