69 /**
70 * @brief Get participant by id
71 *
72 * @param id Participant id.
73 * @return Optional reference to a diagram element.
74 */
75 template <typename T>
76 common::optional_ref<T> get_participant(
77 common::model::diagram_element::id_t id) const
78 {
79 if (participants_.find(id) == participants_.end()) {
80 return {};
81 }
82
83 return common::optional_ref<T>(
84 static_cast<T *>(participants_.at(id).get()));
85 }

231private:
232 /**
233 * This method checks the last messages in sequence (current_messages),
234 * if they represent a block sequence identified by statement_begin
235 * (e.g. if/else) and there are no actual call expressions within this block
236 * statement the entire block statement is removed from the end of the
237 * sequence.
238 *
239 * Otherwise the block statement is ended with a proper statement
240 * (e.g. endif)
241 *
242 * @param m Message to add to the sequence
243 * @param statement_begin Type of message which begins this type of block
244 * statement (e.g. message_t::kIf)
245 * @param current_messages Reference to the sequence messages which should
246 * be amended
247 */
248 void fold_or_end_block_statement(message &&m,
249 common::model::message_t statement_begin,
250 std::vector<message> &current_messages) const;
251
252 bool is_begin_block_message(common::model::message_t mt)
253 {
254 using common::model::message_t;
255 static std::set<message_t> block_begin_types{message_t::kIf,
256 message_t::kWhile, message_t::kDo, message_t::kFor, message_t::kTry,
257 message_t::kSwitch, message_t::kConditional};
258
259 return block_begin_types.count(mt) > 0;
260 };
261
262 bool is_end_block_message(common::model::message_t mt)
263 {
264 using common::model::message_t;
265 static std::set<message_t> block_end_types{message_t::kIfEnd,
266 message_t::kWhileEnd, message_t::kDoEnd, message_t::kForEnd,
267 message_t::kTryEnd, message_t::kSwitchEnd,
268 message_t::kConditionalEnd};
269
270 return block_end_types.count(mt) > 0;
271 };
