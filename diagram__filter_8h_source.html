247/**
248 * @brief Common template for filters involving traversing relationship graph.
249 *
250 * This class template provides a common implementation of a diagram
251 * relationship graph traversal. It is used for filters, which need to check
252 * for instance, whether an element is in some kind of relationship with other
253 * element.
254 *
255 * @tparam DiagramT Diagram type
256 * @tparam ElementT Element type
257 * @tparam ConfigEntryT Type of configuration option used to specify initial
258 * elements for traversal
259 * @tparam MatchOverrideT Type of the matched element
260 */
261template <typename DiagramT, typename ElementT,
262 typename ConfigEntryT = std::string,
263 typename MatchOverrideT = common::model::element>
264struct edge_traversal_filter : public filter_visitor {
265 edge_traversal_filter(filter_t type, relationship_t relationship,
266 std::vector<ConfigEntryT> roots, bool forward = false)
267 : filter_visitor{type}
268 , roots_{std::move(roots)}
269 , relationship_{relationship}
270 , forward_{forward}
271 {
272 }
273
274 ~edge_traversal_filter() override = default;
275
276 tvl::value_t match(const diagram &d, const MatchOverrideT &e) const override
277 {
278 // This filter should only be run only on diagram models after the
279 // entire AST has been visited
280 if (!d.complete())
281 return {};
282
283 if (!check_diagram_type<DiagramT>(d.type()))
284 return {};
285
286 if (roots_.empty())
287 return {};
288
289 const auto &cd = dynamic_cast<const DiagramT &>(d);
290
291 // Calculate the set of matching elements
292 init(cd);
293
294 const auto &fn = e.full_name(false);
295 auto element_ref = detail::get<ElementT>(cd, fn);
296
297 if (!element_ref.has_value())
298 // Couldn't find the element in the diagram
299 return false;
300
301 // Now check if the e element is contained in the calculated set
302 return std::any_of(matching_elements_.begin(), matching_elements_.end(),
303 [&e](const auto &te) {
304 return te.get().full_name(false) == e.full_name(false);
305 });
306 }
307
308private:
309 template <typename C, typename D>
310 bool add_adjacent(const C &from, const D &to,
311 const std::vector<relationship_t> &relationships) const
312 {
313 bool added_new_element{false};
314
315 for (const auto &from_el : from) {
316 // Check if any of its relationships of type relationship_
317 // points to an element already in the matching_elements_
318 // set
319 for (const auto &rel : from_el.get().relationships()) {
320 // Consider only if connected by one of specified relationships
321 if (util::contains(relationships, rel.type())) {
322 for (const auto &to_el : to) {
323 if (rel.destination() ==
324 detail::destination_comparator(to_el.get())) {
325 const auto &to_add = forward_ ? to_el : from_el;
326 if (matching_elements_.insert(to_add).second)
327 added_new_element = true;
328 }
329 }
330 }
331 }
332 }
333
334 return added_new_element;
335 }
336
337 void add_parents(const DiagramT &cd) const
338 {
339 decltype(matching_elements_) parents;
340
341 util::for_each(
342 matching_elements_, [&cd, &parents](const auto &element) {
343 auto parent = detail::get<ElementT, DiagramT>(
344 cd, element.get().path().to_string());
345
346 while (parent.has_value()) {
347 parents.emplace(parent.value());
348 parent = detail::get<ElementT, DiagramT>(
349 cd, parent.value().path().to_string());
350 }
351 });
352
353 matching_elements_.insert(std::begin(parents), std::end(parents));
354 }
355
356 void init(const DiagramT &cd) const
357 {
358 if (initialized_)
359 return;
360
361 // First get all elements specified in the filter configuration
362 // which will serve as starting points for the search
363 // of matching elements
364 for (const auto &root_pattern : roots_) {
365 if constexpr (std::is_same_v<ConfigEntryT,
366 common::string_or_regex>) {
367 auto root_refs = cd.template find<ElementT>(root_pattern);
368
369 for (auto &root : root_refs) {
370 if (root.has_value())
371 matching_elements_.emplace(root.value());
372 }
373 }
374 else {
375 auto root_ref = detail::get<ElementT>(cd, root_pattern);
376 if (root_ref.has_value()) {
377 matching_elements_.emplace(root_ref.value());
378 }
379 }
380 }
381
382 assert(roots_.empty() == matching_elements_.empty());
383
384 bool keep_looking{true};
385 while (keep_looking) {
386 keep_looking = false;
387 if (forward_) {
388 if (add_adjacent(matching_elements_, detail::view<ElementT>(cd),
389 {relationship_}))
390 keep_looking = true;
391 }
392 else {
393 if (add_adjacent(detail::view<ElementT>(cd), matching_elements_,
394 {relationship_}))
395 keep_looking = true;
396 }
397 }
398
399 // For nested diagrams, include also parent elements
400 if ((type() == filter_t::kInclusive) &&
401 (cd.type() == common::model::diagram_t::kPackage)) {
402 add_parents(cd);
403 }
404
405 initialized_ = true;
406 }
407
408 std::vector<ConfigEntryT> roots_;
409 relationship_t relationship_;
410 mutable bool initialized_{false};
411 mutable clanguml::common::reference_set<ElementT> matching_elements_;
412 bool forward_;
413};

461 template <typename ElementT>
462 void find_elements_in_direct_relationship(const diagram &d,
463 std::set<id_t> &effective_context,
464 std::set<clanguml::common::id_t> &current_iteration_context) const
465 {
466 static_assert(std::is_same_v<ElementT, class_diagram::model::class_> ||
467 std::is_same_v<ElementT, class_diagram::model::concept_>,
468 "ElementT must be either class_ or concept_");
469
470 const auto &cd = dynamic_cast<const class_diagram::model::diagram &>(d);
471
472 for (const auto &el : cd.elements<ElementT>()) {
473 for (const relationship &rel : el.get().relationships()) {
474 for (const auto &ec : effective_context) {
475 if (d.should_include(rel.type()) && rel.destination() == ec)
476 current_iteration_context.emplace(el.get().id());
477 }
478 }
479
480 for (const auto &ec : effective_context) {
481 const auto &maybe_concept = cd.find<ElementT>(ec);
482
483 if (!maybe_concept)
484 continue;
485
486 for (const relationship &rel :
487 maybe_concept.value().relationships()) {
488
489 if (d.should_include(rel.type()) &&
490 rel.destination() == el.get().id())
491 current_iteration_context.emplace(el.get().id());
492 }
493 }
494 }
495 }

606 /**
607 * Generic `should_include` overload for various diagram elements.
608 *
609 * @tparam T Type to to match - must match one of filter_visitor's match(T)
610 * @param e Value of type T to match
611 * @return Match result.
612 */
613 template <typename T> bool should_include(const T &e) const
614 {
615 auto exc = tvl::any_of(exclusive_.begin(), exclusive_.end(),
616 [this, &e](const auto &ex) { return ex->match(diagram_, e); });
617
618 if (tvl::is_true(exc))
619 return false;
620
621 auto inc = tvl::all_of(inclusive_.begin(), inclusive_.end(),
622 [this, &e](const auto &in) { return in->match(diagram_, e); });
623
624 return static_cast<bool>(tvl::is_undefined(inc) || tvl::is_true(inc));
625 }
626
627private:
628 /**
629 * @brief Initialize filters.
630 *
631 * Some filters require initialization.
632 *
633 * @param c Diagram config.
634 */
635 void init_filters(const config::diagram &c);
636
637 /*! List of inclusive filters */
638 std::vector<std::unique_ptr<filter_visitor>> inclusive_;
639
640 /*! List of exclusive filters */
641 std::vector<std::unique_ptr<filter_visitor>> exclusive_;
642
643 /*! Reference to the diagram model */
644 const common::model::diagram &diagram_;
645};
