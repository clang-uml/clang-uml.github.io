39bool translation_unit_visitor::VisitNamespaceDecl(clang::NamespaceDecl *ns)
40{
41 assert(ns != nullptr);
42
43 if (config().package_type() == config::package_type_t::kDirectory)
44 return true;
45
46 if (ns->isAnonymousNamespace() || ns->isInline())
47 return true;
48
49 LOG_DBG("= Visiting namespace declaration {} at {}",
50 ns->getQualifiedNameAsString(),
51 ns->getLocation().printToString(source_manager()));
52
53 auto package_path = namespace_{common::get_qualified_name(*ns)};
54 auto package_parent = package_path;
55
56 std::string name;
57 if (!package_path.is_empty())
58 name = package_path.name();
59
60 if (!package_parent.is_empty())
61 package_parent.pop_back();
62
63 const auto usn = config().using_namespace();
64
65 auto p = std::make_unique<common::model::package>(usn);
66 package_path = package_path.relative_to(usn);
67
68 p->set_name(name);
69 p->set_namespace(package_parent);
70 p->set_id(common::to_id(*ns));
71 id_mapper().add(ns->getID(), p->id());
72
73 if (diagram().should_include(*p) && !diagram().get(p->id())) {
74 process_comment(*ns, *p);
75 set_source_location(*ns, *p);
76
77 p->set_style(p->style_spec());
78
79 for (const auto *attr : ns->attrs()) {
80 if (attr->getKind() == clang::attr::Kind::Deprecated) {
81 p->set_deprecated(true);
82 break;
83 }
84 }
85
86 if (!p->skip()) {
87 diagram().add(package_path, std::move(p));
88 }
89 }
90
91 return true;
92}

94bool translation_unit_visitor::VisitEnumDecl(clang::EnumDecl *enm)
95{
96 assert(enm != nullptr);
97
98 // Anonymous enum values should be rendered as class fields
99 // with type enum
100 if (enm->getNameAsString().empty())
101 return true;
102
103 if (!should_include(enm))
104 return true;
105
106 LOG_DBG("= Visiting enum declaration {} at {}",
107 enm->getQualifiedNameAsString(),
108 enm->getLocation().printToString(source_manager()));
109
110 auto e_ptr = std::make_unique<enum_>(config_.using_namespace());
111 auto &e = *e_ptr;
112
113 std::string qualified_name = common::get_qualified_name(*enm);
114
115 auto ns{common::get_tag_namespace(*enm)};
116
117 const auto *parent = enm->getParent();
118
119 std::optional<common::model::diagram_element::id_t> id_opt;
120
121 if (parent != nullptr) {
122 const auto *parent_record_decl =
123 clang::dyn_cast<clang::RecordDecl>(parent);
124
125 if (parent_record_decl != nullptr) {
126 int64_t local_id = parent_record_decl->getID();
127
128 // First check if the parent has been added to the diagram as
129 // regular class
130 id_opt = id_mapper().get_global_id(local_id);
131
132 // If not, check if the parent template declaration is in the model
133 if (!id_opt) {
134 if (parent_record_decl->getDescribedTemplate() != nullptr) {
135 local_id =
136 parent_record_decl->getDescribedTemplate()->getID();
137 id_opt = id_mapper().get_global_id(local_id);
138 }
139 }
140 }
141 }
142
143 if (id_opt && diagram().find<class_>(*id_opt)) {
144 auto parent_class = diagram().find<class_>(*id_opt);
145
146 e.set_namespace(ns);
147 e.set_name(parent_class.value().name() + "##" + enm->getNameAsString());
148 e.set_id(common::to_id(e.full_name(false)));
149 e.add_relationship({relationship_t::kContainment, *id_opt});
150 e.nested(true);
151 }
152 else {
153 e.set_name(common::get_tag_name(*enm));
154 e.set_namespace(ns);
155 e.set_id(common::to_id(e.full_name(false)));
156 }
157
158 id_mapper().add(enm->getID(), e.id());
159
160 process_comment(*enm, e);
161 set_source_location(*enm, e);
162
163 if (e.skip())
164 return true;
165
166 e.set_style(e.style_spec());
167
168 for (const auto &ev : enm->enumerators()) {
169 e.constants().push_back(ev->getNameAsString());
170 }
171
172 add_enum(std::move(e_ptr));
173
174 return true;
175}

177bool translation_unit_visitor::VisitClassTemplateSpecializationDecl(
178 clang::ClassTemplateSpecializationDecl *cls)
179{
180 if (!should_include(cls))
181 return true;
182
183 LOG_DBG("= Visiting template specialization declaration {} at {} "
184 "(described class id {})",
185 cls->getQualifiedNameAsString(),
186 cls->getLocation().printToString(source_manager()),
187 cls->getSpecializedTemplate()
188 ? cls->getSpecializedTemplate()->getTemplatedDecl()->getID()
189 : 0);
190
191 // TODO: Add support for classes defined in function/method bodies
192 if (cls->isLocalClass() != nullptr)
193 return true;
194
195 auto template_specialization_ptr = process_template_specialization(cls);
196
197 if (!template_specialization_ptr)
198 return true;
199
200 auto &template_specialization = *template_specialization_ptr;
201
202 if (cls->hasBody()) {
203 process_template_specialization_children(cls, template_specialization);
204 }
205
206 if (cls->hasDefinition()) {
207 // Process template specialization bases
208 process_class_bases(cls, template_specialization);
209
210 // Process class child entities
211 process_class_children(cls, template_specialization);
212 }
213
214 if (!template_specialization.template_specialization_found()) {
215 // Only do this if we haven't found a bettern specialization during
216 // construction of the template specialization
217 const auto maybe_id =
218 id_mapper().get_global_id(cls->getSpecializedTemplate()->getID());
219 if (maybe_id.has_value())
220 template_specialization.add_relationship(
221 {relationship_t::kInstantiation, maybe_id.value()});
222 }
223
224 if (diagram().should_include(template_specialization)) {
225 const auto full_name = template_specialization.full_name(false);
226 const auto id = template_specialization.id();
227
228 LOG_DBG("Adding class template specialization {} with id {}", full_name,
229 id);
230
231 add_class(std::move(template_specialization_ptr));
232 }
233
234 return true;
235}

273bool translation_unit_visitor::VisitClassTemplateDecl(
274 clang::ClassTemplateDecl *cls)
275{
276 if (!should_include(cls))
277 return true;
278
279 LOG_DBG("= Visiting class template declaration {} at {}",
280 cls->getQualifiedNameAsString(),
281 cls->getLocation().printToString(source_manager()));
282
283 auto c_ptr = create_class_declaration(cls->getTemplatedDecl());
284
285 if (!c_ptr)
286 return true;
287
288 add_processed_template_class(cls->getQualifiedNameAsString());
289
290 // Override the id with the template id, for now we don't care about the
291 // underlying templated class id
292
293 process_template_parameters(*cls, *c_ptr, *c_ptr);
294
295 const auto cls_full_name = c_ptr->full_name(false);
296 const auto id = common::to_id(cls_full_name);
297
298 c_ptr->set_id(id);
299 c_ptr->is_template(true);
300
301 id_mapper().add(cls->getID(), id);
302
303 constexpr auto kMaxConstraintCount = 24U;
304 llvm::SmallVector<const clang::Expr *, kMaxConstraintCount> constraints{};
305 if (cls->hasAssociatedConstraints()) {
306 cls->getAssociatedConstraints(constraints);
307 }
308
309 for (const auto *expr : constraints) {
310 find_relationships_in_constraint_expression(*c_ptr, expr);
311 }
312
313 if (!cls->getTemplatedDecl()->isCompleteDefinition()) {
314 forward_declarations_.emplace(id, std::move(c_ptr));
315 return true;
316 }
317 process_class_declaration(*cls->getTemplatedDecl(), *c_ptr);
318 forward_declarations_.erase(id);
319
320 if (diagram().should_include(*c_ptr)) {
321 const auto name = c_ptr->full_name();
322 LOG_DBG("Adding class template {} with id {}", name, id);
323
324 add_class(std::move(c_ptr));
325 }
326
327 return true;

330bool translation_unit_visitor::VisitRecordDecl(clang::RecordDecl *rec)
331{
332 if (clang::dyn_cast_or_null<clang::CXXRecordDecl>(rec) != nullptr)
333 // This is handled by VisitCXXRecordDecl()
334 return true;
335
336 // It seems we are in a C (not C++) translation unit
337 if (!should_include(rec))
338 return true;
339
340 LOG_DBG("= Visiting record declaration {} at {}",
341 rec->getQualifiedNameAsString(),
342 rec->getLocation().printToString(source_manager()));
343
344 auto record_ptr = create_record_declaration(rec);
345
346 if (!record_ptr)
347 return true;
348
349 const auto rec_id = record_ptr->id();
350
351 id_mapper().add(rec->getID(), rec_id);
352
353 auto &record_model = diagram().find<class_>(rec_id).has_value()
354 ? *diagram().find<class_>(rec_id).get()
355 : *record_ptr;
356
357 if (rec->isCompleteDefinition() && !record_model.complete()) {
358 process_record_members(rec, record_model);
359 record_model.complete(true);
360 }
361
362 auto id = record_model.id();
363 if (!rec->isCompleteDefinition()) {
364 forward_declarations_.emplace(id, std::move(record_ptr));
365 return true;
366 }
367 forward_declarations_.erase(id);
368
369 if (diagram().should_include(record_model)) {
370 LOG_DBG("Adding struct/union {} with id {}",
371 record_model.full_name(false), record_model.id());
372
373 add_class(std::move(record_ptr));
374 }
375 else {
376 LOG_DBG("Skipping struct/union {} with id {}", record_model.full_name(),
377 record_model.id());
378 }
379
380 return true;
381}

383bool translation_unit_visitor::TraverseConceptDecl(clang::ConceptDecl *cpt)
384{
385 if (!should_include(cpt))
386 return true;
387
388 LOG_DBG("= Visiting concept (isType: {}) declaration {} at {}",
389 cpt->isTypeConcept(), cpt->getQualifiedNameAsString(),
390 cpt->getLocation().printToString(source_manager()));
391
392 auto concept_model = create_concept_declaration(cpt);
393
394 if (!concept_model)
395 return true;
396
397 const auto concept_id = concept_model->id();
398
399 id_mapper().add(cpt->getID(), concept_id);
400
401 process_template_parameters(*cpt, *concept_model);
402
403 constexpr auto kMaxConstraintCount = 24U;
404 llvm::SmallVector<const clang::Expr *, kMaxConstraintCount> constraints{};
405 if (cpt->hasAssociatedConstraints()) {
406 cpt->getAssociatedConstraints(constraints);
407 }
408
409 for (const auto *expr : constraints) {
410 find_relationships_in_constraint_expression(*concept_model, expr);
411 }
412
413 if (cpt->getConstraintExpr() != nullptr) {
414 process_constraint_requirements(
415 cpt, cpt->getConstraintExpr(), *concept_model);
416
417 find_relationships_in_constraint_expression(
418 *concept_model, cpt->getConstraintExpr());
419 }
420
421 if (diagram().should_include(*concept_model)) {
422 LOG_DBG("Adding concept {} with id {}", concept_model->full_name(false),
423 concept_model->id());
424
425 add_concept(std::move(concept_model));
426 }
427 else {
428 LOG_DBG("Skipping concept {} with id {}", concept_model->full_name(),
429 concept_model->id());
430 }
431
432 return true;
433}

435void translation_unit_visitor::process_constraint_requirements(
436 const clang::ConceptDecl *cpt, const clang::Expr *expr,
437 model::concept_ &concept_model) const
438{
439 if (const auto *constraint = llvm::dyn_cast<clang::RequiresExpr>(expr);
440 constraint) {
441
442 auto constraint_source = common::to_string(constraint);
443
444 LOG_DBG("== Processing constraint: '{}'", constraint_source);
445
446 for (const auto *requirement : constraint->getRequirements()) {
447 LOG_DBG("== Processing requirement: '{}'", requirement->getKind());
448 }
449
450 // process 'requires (...)' declaration
451 for (const auto *decl : constraint->getBody()->decls()) {
452 if (const auto *parm_var_decl =
453 llvm::dyn_cast<clang::ParmVarDecl>(decl);
454 parm_var_decl) {
455 parm_var_decl->getQualifiedNameAsString();
456
457 auto param_name = parm_var_decl->getQualifiedNameAsString();
458 auto param_type = common::to_string(
459 parm_var_decl->getType(), cpt->getASTContext());
460
461 LOG_DBG("=== Processing parameter variable declaration: {}, {}",
462 param_name, param_type);
463
464 concept_model.add_parameter(
465 {std::move(param_type), std::move(param_name)});
466 }
467 else {
468 LOG_DBG("=== Processing some other concept declaration: {}",
469 decl->getID());
470 }
471 }
472
473 // process concept body requirements '{ }' if any
474 for (const auto *req : constraint->getRequirements()) {
475 if (req->getKind() == clang::concepts::Requirement::RK_Simple) {
476 const auto *simple_req =
477 llvm::dyn_cast<clang::concepts::ExprRequirement>(req);
478
479 if (simple_req != nullptr) {
480 util::if_not_null(
481 simple_req->getExpr(), [&concept_model](const auto *e) {
482 auto simple_expr = common::to_string(e);
483
484 LOG_DBG("=== Processing expression requirement: {}",
485 simple_expr);
486
487 concept_model.add_statement(std::move(simple_expr));
488 });
489 }
490 }
491 else if (req->getKind() == clang::concepts::Requirement::RK_Type) {
492 util::if_not_null(
493 llvm::dyn_cast<clang::concepts::TypeRequirement>(req),
494 [&concept_model, cpt](const auto *t) {
495 auto type_name = common::to_string(
496 t->getType()->getType(), cpt->getASTContext());
497
498 LOG_DBG(
499 "=== Processing type requirement: {}", type_name);
500
501 concept_model.add_statement(std::move(type_name));
502 });
503 }
504 else if (req->getKind() ==
505 clang::concepts::Requirement::RK_Nested) {
506 const auto *nested_req =
507 llvm::dyn_cast<clang::concepts::NestedRequirement>(req);
508
509 if (nested_req != nullptr) {
510 util::if_not_null(
511 nested_req->getConstraintExpr(), [](const auto *e) {
512 LOG_DBG("=== Processing nested requirement: {}",
513 common::to_string(e));
514 });
515 }
516 }
517 else if (req->getKind() ==
518 clang::concepts::Requirement::RK_Compound) {
519 const auto *compound_req =
520 llvm::dyn_cast<clang::concepts::ExprRequirement>(req);
521
522 if (compound_req != nullptr) {
523 const auto *compound_expr_ptr = compound_req->getExpr();
524
525 if (compound_expr_ptr != nullptr) {
526 auto compound_expr =
527 common::to_string(compound_expr_ptr);
528
529 auto req_return_type =
530 compound_req->getReturnTypeRequirement();
531
532 if (!req_return_type.isEmpty()) {
533 compound_expr =
534 fmt::format("{{{}}} -> {}", compound_expr,
535 common::to_string(
536 req_return_type.getTypeConstraint()));
537 }
538 else if (compound_req->hasNoexceptRequirement()) {
539 compound_expr =
540 fmt::format("{{{}}} noexcept", compound_expr);
541 }
542
543 LOG_DBG("=== Processing compound requirement: {}",
544 compound_expr);
545
546 concept_model.add_statement(std::move(compound_expr));
547 }
548 }
549 }
550 }
551 }
552 else if (const auto *binop = llvm::dyn_cast<clang::BinaryOperator>(expr);
553 binop) {
554 process_constraint_requirements(cpt, binop->getLHS(), concept_model);
555 process_constraint_requirements(cpt, binop->getRHS(), concept_model);
556 }
557 else if (const auto *unop = llvm::dyn_cast<clang::UnaryOperator>(expr);
558 unop) {
559 process_constraint_requirements(cpt, unop->getSubExpr(), concept_model);
560 }
561}

563void translation_unit_visitor::find_relationships_in_constraint_expression(
564 clanguml::common::model::element &c, const clang::Expr *expr)
565{
566 if (expr == nullptr)
567 return;
568 found_relationships_t relationships;
569
570 common::if_dyn_cast<clang::UnresolvedLookupExpr>(expr, [&](const auto *ul) {
571 for (const auto ta : ul->template_arguments()) {
572 find_relationships(ta.getArgument().getAsType(), relationships,
573 relationship_t::kConstraint);
574 }
575 });
576
577 common::if_dyn_cast<clang::ConceptSpecializationExpr>(
578 expr, [&](const auto *cs) {
579 process_concept_specialization_relationships(c, cs);
580 });
581
582 common::if_dyn_cast<clang::RequiresExpr>(expr, [&](const auto *re) {
583 // TODO
584 });
585
586 common::if_dyn_cast<clang::BinaryOperator>(expr, [&](const auto *op) {
587 find_relationships_in_constraint_expression(c, op->getLHS());
588 find_relationships_in_constraint_expression(c, op->getRHS());
589 });
590
591 common::if_dyn_cast<clang::UnaryOperator>(expr, [&](const auto *op) {
592 find_relationships_in_constraint_expression(c, op->getSubExpr());
593 });
594
595 for (const auto &[type_element_id, relationship_type] : relationships) {
596 if (type_element_id != c.id() &&
597 (relationship_type != relationship_t::kNone)) {
598
599 relationship r{relationship_type, type_element_id};
600
601 c.add_relationship(std::move(r));
602 }
603 }

606void translation_unit_visitor::process_concept_specialization_relationships(
607 common::model::element &c,
608 const clang::ConceptSpecializationExpr *concept_specialization)
609{
610
611 if (const auto *cpt = concept_specialization->getNamedConcept();
612 should_include(cpt)) {
613
614 const auto cpt_name = cpt->getNameAsString();
615
616 const auto maybe_id = id_mapper().get_global_id(cpt->getID());
617 if (!maybe_id)
618 return;
619
620 const auto target_id = maybe_id.value();
621
622 std::vector<std::string> constrained_template_params;
623
624 size_t argument_index{};
625
626 for (const auto ta : concept_specialization->getTemplateArguments()) {
627 if (ta.getKind() == clang::TemplateArgument::Type) {
628 auto type_name =
629 common::to_string(ta.getAsType(), cpt->getASTContext());
630 extract_constrained_template_param_name(concept_specialization,
631 cpt, constrained_template_params, argument_index,
632 type_name);
633 }
634 else if (ta.getKind() == clang::TemplateArgument::Pack) {
635 if (!ta.getPackAsArray().empty() &&
636 ta.getPackAsArray().front().isPackExpansion()) {
637 const auto &pack_head =
638 ta.getPackAsArray().front().getAsType();
639 auto type_name =
640 common::to_string(pack_head, cpt->getASTContext());
641 extract_constrained_template_param_name(
642 concept_specialization, cpt,
643 constrained_template_params, argument_index, type_name);
644 }
645 }
646 else {
647 auto type_name =
648 common::to_string(ta.getAsType(), cpt->getASTContext());
649 LOG_DBG(
650 "=== Unsupported concept type parameter: {}", type_name);
651 }
652 argument_index++;
653 }
654
655 if (!constrained_template_params.empty())
656 c.add_relationship(
657 {relationship_t::kConstraint, target_id, access_t::kNone,
658 fmt::format(
659 "{}", fmt::join(constrained_template_params, ","))});
660 }

663bool translation_unit_visitor::VisitCXXRecordDecl(clang::CXXRecordDecl *cls)
664{
665 // Skip system headers
666 if (source_manager().isInSystemHeader(cls->getSourceRange().getBegin()))
667 return true;
668
669 if (!should_include(cls))
670 return true;
671
672 LOG_DBG("= Visiting class declaration {} at {}",
673 cls->getQualifiedNameAsString(),
674 cls->getLocation().printToString(source_manager()));
675
676 LOG_DBG(
677 "== getQualifiedNameAsString() = {}", cls->getQualifiedNameAsString());
678 LOG_DBG("== getID() = {}", cls->getID());
679 LOG_DBG("== isTemplateDecl() = {}", cls->isTemplateDecl());
680 LOG_DBG("== isTemplated() = {}", cls->isTemplated());
681 LOG_DBG("== getParent()->isRecord()() = {}", cls->getParent()->isRecord());
682
683 if (const auto *parent_record =
684 clang::dyn_cast<clang::RecordDecl>(cls->getParent());
685 parent_record != nullptr) {
686 LOG_DBG("== getParent()->getQualifiedNameAsString() = {}",
687 parent_record->getQualifiedNameAsString());
688 }
689
690 if (has_processed_template_class(cls->getQualifiedNameAsString()))
691 // If we have already processed the template of this class
692 // skip it
693 return true;
694
695 if (cls->isTemplated() && (cls->getDescribedTemplate() != nullptr)) {
696 // If the described templated of this class is already in the model
697 // skip it:
698 if (id_mapper().get_global_id(cls->getDescribedTemplate()->getID()))
699 return true;
700 }
701
702 // TODO: Add support for classes defined in function/method bodies
703 if (cls->isLocalClass() != nullptr)
704 return true;
705
706 auto c_ptr = create_class_declaration(cls);
707
708 if (!c_ptr)
709 return true;
710
711 const auto cls_id = c_ptr->id();
712
713 id_mapper().add(cls->getID(), cls_id);
714
715 auto &class_model = diagram().find<class_>(cls_id).has_value()
716 ? *diagram().find<class_>(cls_id).get()
717 : *c_ptr;
718
719 if (cls->isCompleteDefinition() && !class_model.complete())
720 process_class_declaration(*cls, class_model);
721
722 auto id = class_model.id();
723 if (!cls->isCompleteDefinition()) {
724 forward_declarations_.emplace(id, std::move(c_ptr));
725 return true;
726 }
727 forward_declarations_.erase(id);
728
729 if (diagram().should_include(class_model)) {
730 LOG_DBG("Adding class {} with id {}", class_model.full_name(false),
731 class_model.id());
732
733 add_class(std::move(c_ptr));
734 }
735 else {
736 LOG_DBG("Skipping class {} with id {}", class_model.full_name(),
737 class_model.id());
738 }
739
740 return true;
741}

851void translation_unit_visitor::process_record_parent(
852 clang::RecordDecl *cls, class_ &c, const namespace_ &ns)
853{
854 const auto *parent = cls->getParent();
855
856 std::optional<common::model::diagram_element::id_t> id_opt;
857
858 auto parent_ns = ns;
859 if (parent != nullptr) {
860 const auto *parent_record_decl =
861 clang::dyn_cast<clang::RecordDecl>(parent);
862
863 if (parent_record_decl != nullptr) {
864 parent_ns = common::get_tag_namespace(*parent_record_decl);
865
866 int64_t local_id = parent_record_decl->getID();
867
868 // First check if the parent has been added to the diagram as
869 // regular class
870 id_opt = id_mapper().get_global_id(local_id);
871
872 // If not, check if the parent template declaration is in the
873 // model
874 if (!id_opt) {
875 if (parent_record_decl->getDescribedTemplate() != nullptr) {
876 local_id =
877 parent_record_decl->getDescribedTemplate()->getID();
878 id_opt = id_mapper().get_global_id(local_id);
879 }
880 }
881 }
882 }
883
884 if (id_opt && diagram_.find<class_>(*id_opt)) {
885 // Here we have 2 options, either:
886 // - the parent is a regular C++ class/struct
887 // - the parent is a class template declaration/specialization
888 auto parent_class = diagram_.find<class_>(*id_opt);
889
890 c.set_namespace(parent_ns);
891 const auto cls_name = cls->getNameAsString();
892 if (cls_name.empty()) {
893 // Nested structs can be anonymous
894 if (anonymous_struct_relationships_.count(cls->getID()) > 0) {
895 const auto &[label, hint, access] =
896 anonymous_struct_relationships_[cls->getID()];
897
898 c.set_name(parent_class.value().name() + "##" +
899 fmt::format("({})", label));
900
901 parent_class.value().add_relationship(
902 {hint, common::to_id(c.full_name(false)), access, label});
903 }
904 else
905 c.set_name(parent_class.value().name() + "##" +
906 fmt::format(
907 "(anonymous_{})", std::to_string(cls->getID())));
908 }
909 else {
910 c.set_name(
911 parent_class.value().name() + "##" + cls->getNameAsString());
912 }
913
914 c.set_id(common::to_id(c.full_name(false)));
915
916 if (!cls->getNameAsString().empty()) {
917 // Don't add anonymous structs as contained in the class
918 // as they are already added as aggregations
919 c.add_relationship({relationship_t::kContainment, *id_opt});
920 }
921
922 c.nested(true);
923 }

938bool translation_unit_visitor::process_template_parameters(
939 const clang::TemplateDecl &template_declaration,
940 common::model::template_trait &c,
941 common::optional_ref<common::model::element> templated_element)
942{
943 LOG_DBG("Processing {} template parameters...",
944 common::get_qualified_name(template_declaration));
945
946 if (template_declaration.getTemplateParameters() == nullptr)
947 return false;
948
949 for (const auto *parameter :
950 *template_declaration.getTemplateParameters()) {
951 if (clang::dyn_cast_or_null<clang::TemplateTypeParmDecl>(parameter) !=
952 nullptr) {
953 const auto *template_type_parameter =
954 clang::dyn_cast_or_null<clang::TemplateTypeParmDecl>(parameter);
955
956 std::optional<std::string> default_arg;
957 if (template_type_parameter->hasDefaultArgument()) {
958 default_arg =
959 template_type_parameter->getDefaultArgument().getAsString();
960 }
961
962 auto parameter_name = template_type_parameter->getNameAsString();
963 if (parameter_name.empty())
964 parameter_name = "typename";
965
966 auto ct = template_parameter::make_template_type(parameter_name,
967 default_arg, template_type_parameter->isParameterPack());
968
969 if (template_type_parameter->getTypeConstraint() != nullptr) {
970 util::if_not_null(template_type_parameter->getTypeConstraint()
971 ->getNamedConcept(),
972 [this, &ct, &templated_element](
973 const clang::ConceptDecl *named_concept) mutable {
974 ct.set_concept_constraint(
975 named_concept->getQualifiedNameAsString());
976 if (templated_element &&
977 should_include(named_concept)) {
978 templated_element.value().add_relationship(
979 {relationship_t::kConstraint,
980 id_mapper()
981 .get_global_id(named_concept->getID())
982 .value(),
983 access_t::kNone, ct.name().value()});
984 }
985 });
986 }
987
988 c.add_template(std::move(ct));
989 }
990 else if (clang::dyn_cast_or_null<clang::NonTypeTemplateParmDecl>(
991 parameter) != nullptr) {
992 const auto *template_nontype_parameter =
993 clang::dyn_cast_or_null<clang::NonTypeTemplateParmDecl>(
994 parameter);
995
996 std::optional<std::string> default_arg;
997
998 if (template_nontype_parameter->hasDefaultArgument())
999 default_arg = common::to_string(
1000 template_nontype_parameter->getDefaultArgument());
1001
1002 auto ct = template_parameter::make_non_type_template(
1003 template_nontype_parameter->getType().getAsString(),
1004 template_nontype_parameter->getNameAsString(), default_arg,
1005 template_nontype_parameter->isParameterPack());
1006
1007 c.add_template(std::move(ct));
1008 }
1009 else if (clang::dyn_cast_or_null<clang::TemplateTemplateParmDecl>(
1010 parameter) != nullptr) {
1011 const auto *template_template_parameter =
1012 clang::dyn_cast_or_null<clang::TemplateTemplateParmDecl>(
1013 parameter);
1014 std::optional<std::string> default_arg;
1015 if (template_template_parameter->hasDefaultArgument()) {
1016 default_arg = common::to_string(
1017 template_template_parameter->getDefaultArgument()
1018 .getArgument());
1019 }
1020 auto ct = template_parameter::make_template_template_type(
1021 template_template_parameter->getNameAsString(), default_arg,
1022 template_template_parameter->isParameterPack());
1023
1024 c.add_template(std::move(ct));
1025 }
1026 else {
1027 // pass
1028 }
1029 }
1030
1031 return false;
1032}

1034void translation_unit_visitor::process_class_bases(
1035 const clang::CXXRecordDecl *cls, class_ &c)
1036{
1037 for (const auto &base : cls->bases()) {
1038 class_parent cp;
1039 auto name_and_ns = common::model::namespace_{
1040 common::to_string(base.getType(), cls->getASTContext())};
1041
1042 cp.set_name(name_and_ns.to_string());
1043
1044 if (const auto *record_type =
1045 base.getType()->getAs<clang::RecordType>();
1046 record_type != nullptr) {
1047 cp.set_name(record_type->getDecl()->getQualifiedNameAsString());
1048 cp.set_id(common::to_id(*record_type->getDecl()));
1049 }
1050 else if (const auto *tsp =
1051 base.getType()->getAs<clang::TemplateSpecializationType>();
1052 tsp != nullptr) {
1053 auto template_specialization_ptr = tbuilder().build(cls, *tsp, {});
1054 if (template_specialization_ptr) {
1055 cp.set_id(template_specialization_ptr->id());
1056 }
1057 }
1058 else
1059 // This could be a template parameter - we don't want it here
1060 continue;
1061
1062 cp.is_virtual(base.isVirtual());
1063
1064 cp.set_access(
1065 common::access_specifier_to_access_t(base.getAccessSpecifier()));
1066
1067 LOG_DBG("Found base class {} [{}] for class {}", cp.name(), cp.id(),
1068 c.name());
1069
1070 c.add_parent(std::move(cp));
1071 }

1074void translation_unit_visitor::process_template_specialization_children(
1075 const clang::ClassTemplateSpecializationDecl *cls, class_ &c)
1076{
1077 assert(cls != nullptr);
1078
1079 // Iterate over class methods (both regular and static)
1080 for (const auto *method : cls->methods()) {
1081 if (method != nullptr) {
1082 process_method(*method, c);
1083 }
1084 }
1085
1086 // Iterate over class template methods
1087 if (const auto *cls_decl_context =
1088 clang::dyn_cast_or_null<clang::DeclContext>(cls);
1089 cls_decl_context != nullptr) {
1090 for (auto const *decl_iterator :
1091 clang::dyn_cast_or_null<clang::DeclContext>(cls)->decls()) {
1092 auto const *method_template =
1093 llvm::dyn_cast_or_null<clang::FunctionTemplateDecl>(
1094 decl_iterator);
1095 if (method_template == nullptr)
1096 continue;
1097
1098 process_template_method(*method_template, c);
1099 }
1100 }
1101
1102 // Iterate over regular class fields
1103 for (const auto *field : cls->fields()) {
1104 if (field != nullptr)
1105 process_field(*field, c);
1106 }
1107
1108 // Static fields have to be processed by iterating over variable
1109 // declarations
1110 for (const auto *decl : cls->decls()) {
1111 if (decl->getKind() == clang::Decl::Var) {
1112 const clang::VarDecl *variable_declaration{
1113 dynamic_cast<const clang::VarDecl *>(decl)};
1114 if ((variable_declaration != nullptr) &&
1115 variable_declaration->isStaticDataMember()) {
1116 process_static_field(*variable_declaration, c);
1117 }
1118 }
1119 else if (decl->getKind() == clang::Decl::Enum) {
1120 const auto *enum_decl =
1121 clang::dyn_cast_or_null<clang::EnumDecl>(decl);
1122 if (enum_decl == nullptr)
1123 continue;
1124
1125 if (enum_decl->getNameAsString().empty()) {
1126 for (const auto *enum_const : enum_decl->enumerators()) {
1127 class_member m{common::access_specifier_to_access_t(
1128 enum_decl->getAccess()),
1129 enum_const->getNameAsString(), "enum"};
1130 c.add_member(std::move(m));
1131 }
1132 }
1133 }
1134 }
1135
1136 if (cls->hasFriends()) {
1137 for (const auto *friend_declaration : cls->friends()) {
1138 process_friend(*friend_declaration, c);
1139 }
1140 }
1141}

1243void translation_unit_visitor::process_method(
1244 const clang::CXXMethodDecl &mf, class_ &c)
1245{
1246 // TODO: For now skip implicitly default methods
1247 // in the future, add config option to choose
1248 if (mf.isDefaulted() && !mf.isExplicitlyDefaulted())
1249 return;
1250
1251 auto method_return_type =
1252 common::to_string(mf.getReturnType(), mf.getASTContext());
1253
1254 ensure_lambda_type_is_relative(method_return_type);
1255
1256 auto method_name = mf.getNameAsString();
1257 if (mf.isTemplated()) {
1258 // Sometimes in template specializations method names contain the
1259 // template parameters for some reason - drop them
1260 // Is there a better way to do this?
1261 method_name = method_name.substr(0, method_name.find('<'));
1262 }
1263
1264 class_method method{common::access_specifier_to_access_t(mf.getAccess()),
1265 util::trim(method_name), method_return_type};
1266
1267 process_method_properties(mf, c, method_name, method);
1268
1269 process_comment(mf, method);
1270
1271 // Register the source location of the field declaration
1272 set_source_location(mf, method);
1273
1274 if (method.skip())
1275 return;
1276
1277 for (const auto *param : mf.parameters()) {
1278 if (param != nullptr)
1279 process_function_parameter(*param, method, c);
1280 }
1281
1282 // find relationship for return type
1283 found_relationships_t relationships;
1284
1285 // Move dereferencing to build() method of template_builder
1286 if (const auto *templ = mf.getReturnType()
1287 .getNonReferenceType()
1288 .getUnqualifiedType()
1289 ->getAs<clang::TemplateSpecializationType>();
1290 templ != nullptr) {
1291 const auto *unaliased_type = templ;
1292 if (unaliased_type->isTypeAlias())
1293 unaliased_type = unaliased_type->getAliasedType()
1294 ->getAs<clang::TemplateSpecializationType>();
1295
1296 if (unaliased_type != nullptr) {
1297 auto template_specialization_ptr = tbuilder().build(
1298 unaliased_type->getTemplateName().getAsTemplateDecl(),
1299 *unaliased_type, &c);
1300
1301 if (diagram().should_include(
1302 template_specialization_ptr->get_namespace())) {
1303 relationships.emplace_back(template_specialization_ptr->id(),
1304 relationship_t::kDependency);
1305
1306 add_class(std::move(template_specialization_ptr));
1307 }
1308 }
1309 }
1310
1311 find_relationships(
1312 mf.getReturnType(), relationships, relationship_t::kDependency);
1313
1314 for (const auto &[type_element_id, relationship_type] : relationships) {
1315 if (type_element_id != c.id() &&
1316 (relationship_type != relationship_t::kNone)) {
1317 relationship r{relationship_t::kDependency, type_element_id};
1318
1319 LOG_DBG("Adding method return type relationship from {}::{} to "
1320 "{}: {}",
1321 c.full_name(), mf.getNameAsString(),
1322 clanguml::common::model::to_string(r.type()), r.label());
1323
1324 c.add_relationship(std::move(r));
1325 }
1326 }
1327
1328 // Also consider the container itself if it is a template
1329 // instantiation it's arguments could count as reference to relevant
1330 // types
1331 auto underlying_type = mf.getReturnType();
1332 if (underlying_type->isReferenceType())
1333 underlying_type = underlying_type.getNonReferenceType();
1334 if (underlying_type->isPointerType())
1335 underlying_type = underlying_type->getPointeeType();
1336
1337 if (const auto *atsp = underlying_type->getAs<clang::AutoType>();
1338 atsp != nullptr) {
1339 process_function_parameter_find_relationships_in_autotype(c, atsp);
1340 }
1341
1342 if (diagram().should_include(method)) {
1343 LOG_DBG("Adding method: {}", method.name());
1344
1345 c.add_method(std::move(method));
1346 }
1347}

1372void translation_unit_visitor::
1373 process_function_parameter_find_relationships_in_autotype(
1374 class_ &c, const clang::AutoType *atsp)
1375{
1376 auto desugared_atsp = atsp->getDeducedType();
1377
1378 if (atsp->isSugared()) {
1379 const auto *deduced_type =
1380 atsp->desugar()->getAs<clang::DeducedTemplateSpecializationType>();
1381
1382 if (deduced_type != nullptr)
1383 desugared_atsp = deduced_type->getDeducedType();
1384 }
1385
1386 if (desugared_atsp.isNull())
1387 return;
1388
1389 const auto *deduced_record_type = desugared_atsp->isRecordType()
1390 ? desugared_atsp->getAs<clang::RecordType>()
1391 : nullptr;
1392
1393 if (deduced_record_type != nullptr) {
1394 if (auto *deduced_auto_decl =
1395 llvm::dyn_cast_or_null<clang::ClassTemplateSpecializationDecl>(
1396 deduced_record_type->getDecl());
1397 deduced_auto_decl != nullptr) {
1398
1399 const auto diagram_class_count_before_visit =
1400 diagram().classes().size();
1401
1402 VisitClassTemplateSpecializationDecl(deduced_auto_decl);
1403
1404 const bool visitor_added_new_template_specialization =
1405 (diagram().classes().size() -
1406 diagram_class_count_before_visit) > 0;
1407
1408 if (visitor_added_new_template_specialization) {
1409 const auto &template_specialization_model =
1410 diagram().classes().back();
1411
1412 if (should_include(deduced_auto_decl)) {
1413 relationship r{relationship_t::kDependency,
1414 template_specialization_model.get().id()};
1415
1416 c.add_relationship(std::move(r));
1417 }
1418 }
1419 }

1423void translation_unit_visitor::process_template_method(
1424 const clang::FunctionTemplateDecl &mf, class_ &c)
1425{
1426 // TODO: For now skip implicitly default methods
1427 // in the future, add config option to choose
1428 if (mf.getTemplatedDecl()->isDefaulted() &&
1429 !mf.getTemplatedDecl()->isExplicitlyDefaulted())
1430 return;
1431
1432 class_method method{common::access_specifier_to_access_t(mf.getAccess()),
1433 util::trim(mf.getNameAsString()),
1434 mf.getTemplatedDecl()->getReturnType().getAsString()};
1435
1436 auto method_name = mf.getNameAsString();
1437 if (mf.isTemplated()) {
1438 // Sometimes in template specializations method names contain the
1439 // template parameters for some reason - drop them
1440 // Is there a better way to do this?
1441 method_name = method_name.substr(0, method_name.find('<'));
1442 }
1443
1444 util::if_not_null(
1445 clang::dyn_cast<clang::CXXMethodDecl>(mf.getTemplatedDecl()),
1446 [&](const auto *decl) {
1447 process_method_properties(*decl, c, method_name, method);
1448 });
1449
1450 process_template_parameters(mf, method);
1451
1452 process_comment(mf, method);
1453
1454 if (method.skip())
1455 return;
1456
1457 for (const auto *param : mf.getTemplatedDecl()->parameters()) {
1458 if (param != nullptr)
1459 process_function_parameter(*param, method, c);
1460 }
1461
1462 if (diagram().should_include(method)) {
1463 LOG_DBG("Adding method: {}", method.name());
1464
1465 c.add_method(std::move(method));
1466 }
1467}

1469bool translation_unit_visitor::find_relationships(const clang::QualType &type,
1470 found_relationships_t &relationships,
1471 clanguml::common::model::relationship_t relationship_hint)
1472{
1473 bool result{false};
1474
1475 if (type->isPointerType()) {
1476 relationship_hint = relationship_t::kAssociation;
1477 find_relationships(
1478 type->getPointeeType(), relationships, relationship_hint);
1479 }
1480 else if (type->isRValueReferenceType()) {
1481 relationship_hint = relationship_t::kAggregation;
1482 find_relationships(
1483 type.getNonReferenceType(), relationships, relationship_hint);
1484 }
1485 else if (type->isLValueReferenceType()) {
1486 relationship_hint = relationship_t::kAssociation;
1487 find_relationships(
1488 type.getNonReferenceType(), relationships, relationship_hint);
1489 }
1490 else if (type->isArrayType()) {
1491 find_relationships(type->getAsArrayTypeUnsafe()->getElementType(),
1492 relationships, relationship_t::kAggregation);
1493 }
1494 else if (type->isEnumeralType()) {
1495 if (const auto *enum_type = type->getAs<clang::EnumType>();
1496 enum_type != nullptr) {
1497 relationships.emplace_back(
1498 common::to_id(*enum_type->getDecl()), relationship_hint);
1499 }
1500 }
1501 else if (type->isRecordType()) {
1502 const auto *type_instantiation_decl =
1503 type->getAs<clang::TemplateSpecializationType>();
1504
1505 if (type_instantiation_decl != nullptr) {
1506 // If this template should be included in the diagram
1507 // add it - and then process recursively its arguments
1508 if (should_include(type_instantiation_decl->getTemplateName()
1509 .getAsTemplateDecl())) {
1510 relationships.emplace_back(
1511 type_instantiation_decl->getTemplateName()
1512 .getAsTemplateDecl()
1513 ->getID(),
1514 relationship_hint);
1515 }
1516 for (const auto &template_argument :
1517 type_instantiation_decl->template_arguments()) {
1518 const auto template_argument_kind = template_argument.getKind();
1519 if (template_argument_kind ==
1520 clang::TemplateArgument::ArgKind::Integral) {
1521 // pass
1522 }
1523 else if (template_argument_kind ==
1524 clang::TemplateArgument::ArgKind::Null) {
1525 // pass
1526 }
1527 else if (template_argument_kind ==
1528 clang::TemplateArgument::ArgKind::Expression) {
1529 // pass
1530 }
1531 else if (template_argument.getKind() ==
1532 clang::TemplateArgument::ArgKind::NullPtr) {
1533 // pass
1534 }
1535 else if (template_argument_kind ==
1536 clang::TemplateArgument::ArgKind::Template) {
1537 // pass
1538 }
1539 else if (template_argument_kind ==
1540 clang::TemplateArgument::ArgKind::TemplateExpansion) {
1541 // pass
1542 }
1543 else if (const auto *function_type =
1544 template_argument.getAsType()
1545 ->getAs<clang::FunctionProtoType>();
1546 function_type != nullptr) {
1547 for (const auto &param_type :
1548 function_type->param_types()) {
1549 result = find_relationships(param_type, relationships,
1550 relationship_t::kDependency);
1551 }
1552 }
1553 else if (template_argument_kind ==
1554 clang::TemplateArgument::ArgKind::Type) {
1555 result = find_relationships(template_argument.getAsType(),
1556 relationships, relationship_hint);
1557 }
1558 }
1559 }
1560 else if (type->getAsCXXRecordDecl() != nullptr) {
1561 const auto target_id = common::to_id(*type->getAsCXXRecordDecl());
1562 relationships.emplace_back(target_id, relationship_hint);
1563 result = true;
1564 }
1565 else {
1566 const auto target_id = common::to_id(*type->getAsRecordDecl());
1567 relationships.emplace_back(target_id, relationship_hint);
1568 result = true;
1569 }
1570 }
1571 else if (const auto *template_specialization_type =
1572 type->getAs<clang::TemplateSpecializationType>();
1573 template_specialization_type != nullptr) {
1574 if (should_include(template_specialization_type->getTemplateName()
1575 .getAsTemplateDecl())) {
1576 relationships.emplace_back(
1577 template_specialization_type->getTemplateName()
1578 .getAsTemplateDecl()
1579 ->getID(),
1580 relationship_hint);
1581 }
1582 for (const auto &template_argument :
1583 template_specialization_type->template_arguments()) {
1584 const auto template_argument_kind = template_argument.getKind();
1585 if (template_argument_kind ==
1586 clang::TemplateArgument::ArgKind::Integral) {
1587 // pass
1588 }
1589 else if (template_argument_kind ==
1590 clang::TemplateArgument::ArgKind::Null) {
1591 // pass
1592 }
1593 else if (template_argument_kind ==
1594 clang::TemplateArgument::ArgKind::Expression) {
1595 // pass
1596 }
1597 else if (template_argument.getKind() ==
1598 clang::TemplateArgument::ArgKind::NullPtr) {
1599 // pass
1600 }
1601 else if (template_argument_kind ==
1602 clang::TemplateArgument::ArgKind::Template) {
1603 // pass
1604 }
1605 else if (template_argument_kind ==
1606 clang::TemplateArgument::ArgKind::TemplateExpansion) {
1607 // pass
1608 }
1609 else if (const auto *function_type =
1610 template_argument.getAsType()
1611 ->getAs<clang::FunctionProtoType>();
1612 function_type != nullptr) {
1613 for (const auto &param_type : function_type->param_types()) {
1614 result = find_relationships(
1615 param_type, relationships, relationship_t::kDependency);
1616 }
1617 }
1618 else if (template_argument_kind ==
1619 clang::TemplateArgument::ArgKind::Type) {
1620 result = find_relationships(template_argument.getAsType(),
1621 relationships, relationship_hint);
1622 }
1623 }
1624 }
1625
1626 return result;
1627}

1629void translation_unit_visitor::process_function_parameter(
1630 const clang::ParmVarDecl &p, class_method &method, class_ &c,
1631 const std::set<std::string> & /*template_parameter_names*/)
1632{
1633 method_parameter parameter;
1634 parameter.set_name(p.getNameAsString());
1635
1636 process_comment(p, parameter);
1637
1638 if (parameter.skip())
1639 return;
1640
1641 auto parameter_type = common::to_string(p.getType(), p.getASTContext());
1642
1643 // Is there no better way to determine that 'type' is a lambda?
1644 ensure_lambda_type_is_relative(parameter_type);
1645
1646 parameter.set_type(parameter_type);
1647
1648 if (p.hasDefaultArg()) {
1649 const auto *default_arg = p.getDefaultArg();
1650 if (default_arg != nullptr) {
1651 auto default_arg_str = common::get_source_text(
1652 default_arg->getSourceRange(), source_manager());
1653 parameter.set_default_value(default_arg_str);
1654 }
1655 }
1656
1657 if (!parameter.skip_relationship()) {
1658 // find relationship for the type
1659 found_relationships_t relationships;
1660
1661 LOG_DBG("Looking for relationships in type: {}",
1662 common::to_string(p.getType(), p.getASTContext()));
1663
1664 if (const auto *templ =
1665 p.getType()
1666 .getNonReferenceType()
1667 .getUnqualifiedType()
1668 ->getAs<clang::TemplateSpecializationType>();
1669 templ != nullptr) {
1670 auto template_specialization_ptr = tbuilder().build(
1671 templ->getTemplateName().getAsTemplateDecl(), *templ, &c);
1672
1673 if (diagram().should_include(
1674 template_specialization_ptr->get_namespace())) {
1675 relationships.emplace_back(template_specialization_ptr->id(),
1676 relationship_t::kDependency);
1677
1678 add_class(std::move(template_specialization_ptr));
1679 }
1680 }
1681
1682 find_relationships(
1683 p.getType(), relationships, relationship_t::kDependency);
1684
1685 for (const auto &[type_element_id, relationship_type] : relationships) {
1686 if (type_element_id != c.id() &&
1687 (relationship_type != relationship_t::kNone)) {
1688 relationship r{relationship_t::kDependency, type_element_id};
1689
1690 LOG_DBG("Adding function parameter relationship from {} to "
1691 "{}: {}",
1692 c.full_name(), clanguml::common::model::to_string(r.type()),
1693 r.label());
1694
1695 c.add_relationship(std::move(r));
1696 }
1697 }
1698 }
1699
1700 method.add_parameter(std::move(parameter));
1701}

1703void translation_unit_visitor::ensure_lambda_type_is_relative(
1704 std::string &parameter_type) const
1705{
1706#ifdef _MSC_VER
1707 auto root_name =
1708 fmt::format("{}", std::filesystem::current_path().root_name().string());
1709#else
1710 auto root_name = std::string{"/"};
1711#endif
1712
1713 std::string lambda_prefix{fmt::format("(lambda at {}", root_name)};
1714
1715 while (parameter_type.find(lambda_prefix) != std::string::npos) {
1716 auto lambda_begin = parameter_type.find(lambda_prefix);
1717 auto lambda_prefix_size = lambda_prefix.size();
1718#ifdef _MSC_VER
1719 // Skip the `\` or `/` after drive letter and semicolon
1720 lambda_prefix_size++;
1721#endif
1722 auto absolute_lambda_path_end =
1723 parameter_type.find(':', lambda_begin + lambda_prefix_size);
1724 auto absolute_lambda_path = parameter_type.substr(
1725 lambda_begin + lambda_prefix_size - 1,
1726 absolute_lambda_path_end - (lambda_begin + lambda_prefix_size - 1));
1727
1728 auto relative_lambda_path = util::path_to_url(
1729 config().make_path_relative(absolute_lambda_path).string());
1730
1731 parameter_type = fmt::format("{}(lambda at {}{}",
1732 parameter_type.substr(0, lambda_begin), relative_lambda_path,
1733 parameter_type.substr(absolute_lambda_path_end));
1734 }
1735}

1737void translation_unit_visitor::add_relationships(class_ &c,
1738 const class_member &field, const found_relationships_t &relationships,
1739 bool break_on_first_aggregation)
1740{
1741 auto [decorator_rtype, decorator_rmult] = field.get_relationship();
1742
1743 for (const auto &[target, relationship_type] : relationships) {
1744 if (relationship_type != relationship_t::kNone) {
1745 relationship r{relationship_type, target};
1746 r.set_label(field.name());
1747 r.set_access(field.access());
1748 if (decorator_rtype != relationship_t::kNone) {
1749 r.set_type(decorator_rtype);
1750 auto mult = util::split(decorator_rmult, ":", false);
1751 if (mult.size() == 2) {
1752 r.set_multiplicity_source(mult[0]);
1753 r.set_multiplicity_destination(mult[1]);
1754 }
1755 }
1756 r.set_style(field.style_spec());
1757
1758 LOG_DBG("Adding relationship from {} to {} with label {}",
1759 c.full_name(false), r.destination(),
1760 clanguml::common::model::to_string(r.type()), r.label());
1761
1762 c.add_relationship(std::move(r));
1763
1764 if (break_on_first_aggregation &&
1765 relationship_type == relationship_t::kAggregation)
1766 break;
1767 }
1768 }

1844void translation_unit_visitor::process_field(
1845 const clang::FieldDecl &field_declaration, class_ &c)
1846{
1847 LOG_DBG(
1848 "== Visiting record member {}", field_declaration.getNameAsString());
1849
1850 // Default hint for relationship is aggregation
1851 auto relationship_hint = relationship_t::kAggregation;
1852 // If the first type of the template instantiation of this field type
1853 // has been added as aggregation relationship with class 'c', don't
1854 // add it's nested template types as aggregation
1855 [[maybe_unused]] bool template_instantiation_added_as_aggregation{false};
1856 // The actual field type
1857 auto field_type = field_declaration.getType();
1858 // String representation of the field type
1859 auto type_name =
1860 common::to_string(field_type, field_declaration.getASTContext());
1861 // The field name
1862 const auto field_name = field_declaration.getNameAsString();
1863
1864 auto field_type_str =
1865 common::to_string(field_type, field_declaration.getASTContext(), false);
1866
1867 ensure_lambda_type_is_relative(field_type_str);
1868
1869 class_member field{
1870 common::access_specifier_to_access_t(field_declaration.getAccess()),
1871 field_name, field_type_str};
1872
1873 // Parse the field comment
1874 process_comment(field_declaration, field);
1875 // Register the source location of the field declaration
1876 set_source_location(field_declaration, field);
1877
1878 // If the comment contains a skip directive, just return
1879 if (field.skip())
1880 return;
1881
1882 if (field_type->isPointerType()) {
1883 relationship_hint = relationship_t::kAssociation;
1884 field_type = field_type->getPointeeType();
1885 }
1886 else if (field_type->isLValueReferenceType()) {
1887 relationship_hint = relationship_t::kAssociation;
1888 field_type = field_type.getNonReferenceType();
1889 }
1890 else if (field_type->isRValueReferenceType()) {
1891 field_type = field_type.getNonReferenceType();
1892 }
1893
1894 if (type_name.find("std::shared_ptr") == 0)
1895 relationship_hint = relationship_t::kAssociation;
1896 if (type_name.find("std::weak_ptr") == 0)
1897 relationship_hint = relationship_t::kAssociation;
1898
1899 found_relationships_t relationships;
1900
1901 const auto *template_field_type =
1902 field_type->getAs<clang::TemplateSpecializationType>();
1903 // TODO: Refactor to an unalias_type() method
1904 if (template_field_type != nullptr)
1905 if (template_field_type->isTypeAlias())
1906 template_field_type =
1907 template_field_type->getAliasedType()
1908 ->getAs<clang::TemplateSpecializationType>();
1909
1910 bool field_type_is_template_template_parameter{false};
1911 if (template_field_type != nullptr) {
1912 // Skip types which are template template parameters of the parent
1913 // template
1914 for (const auto &class_template_param : c.template_params()) {
1915 if (class_template_param.name() ==
1916 template_field_type->getTemplateName()
1917 .getAsTemplateDecl()
1918 ->getNameAsString() +
1919 "<>") {
1920 field_type_is_template_template_parameter = true;
1921 }
1922 }
1923 }
1924
1925 // Process the type which is template instantiation of some sort
1926 if (template_field_type != nullptr &&
1927 !field_type_is_template_template_parameter) {
1928 // Build the template instantiation for the field type
1929 auto template_specialization_ptr = tbuilder().build(
1930 field_type->getAs<clang::TemplateSpecializationType>()
1931 ->getTemplateName()
1932 .getAsTemplateDecl(),
1933 *template_field_type, {&c});
1934
1935 if (!field.skip_relationship() && template_specialization_ptr) {
1936 const auto &template_specialization = *template_specialization_ptr;
1937
1938 // Check if this template instantiation should be added to the
1939 // current diagram. Even if the top level template type for
1940 // this instantiation should not be part of the diagram, e.g.
1941 // it's a std::vector<>, it's nested types might be added
1942 bool add_template_instantiation_to_diagram{false};
1943 if (diagram().should_include(
1944 template_specialization.get_namespace())) {
1945
1946 found_relationships_t::value_type r{
1947 template_specialization.id(), relationship_hint};
1948
1949 add_template_instantiation_to_diagram = true;
1950
1951 // If the template instantiation for the build type has been
1952 // added as aggregation, skip its nested templates
1953 template_instantiation_added_as_aggregation =
1954 relationship_hint == relationship_t::kAggregation;
1955 relationships.emplace_back(std::move(r));
1956 }
1957
1958 // Try to find relationships to types nested in the template
1959 // instantiation
1960 found_relationships_t nested_relationships;
1961 if (!template_instantiation_added_as_aggregation) {
1962 for (const auto &template_argument :
1963 template_specialization.template_params()) {
1964
1965 LOG_DBG("Looking for nested relationships from {}::{} in "
1966 "template argument {}",
1967 c.full_name(false), field_name,
1968 template_argument.to_string(
1969 config().using_namespace(), false));
1970
1971 template_instantiation_added_as_aggregation =
1972 template_argument.find_nested_relationships(
1973 nested_relationships, relationship_hint,
1974 [&d = diagram()](const std::string &full_name) {
1975 if (full_name.empty())
1976 return false;
1977 auto [ns, name] = common::split_ns(full_name);
1978 return d.should_include(ns, name);
1979 });
1980 }
1981
1982 // Add any relationships to the class 'c' to the diagram,
1983 // unless the top level type has been added as aggregation
1984 add_relationships(c, field, nested_relationships,
1985 /* break on first aggregation */ false);
1986 }
1987
1988 // Add the template instantiation object to the diagram if it
1989 // matches the include pattern
1990 if (add_template_instantiation_to_diagram)
1991 add_class(std::move(template_specialization_ptr));
1992 }
1993 }
1994
1995 if (!field.skip_relationship()) {
1996 // Find relationship for the type if the type has not been added
1997 // as aggregation
1998 if (!template_instantiation_added_as_aggregation) {
1999 if ((field_type->getAsRecordDecl() != nullptr) &&
2000 field_type->getAsRecordDecl()->getNameAsString().empty()) {
2001 // Relationships to fields whose type is an anonymous nested
2002 // struct have to be handled separately here
2003 anonymous_struct_relationships_[field_type->getAsRecordDecl()
2004 ->getID()] =
2005 std::make_tuple(
2006 field.name(), relationship_hint, field.access());
2007 }
2008 else
2009 find_relationships(
2010 field_type, relationships, relationship_hint);
2011 }
2012
2013 add_relationships(c, field, relationships);
2014 }
2015
2016 c.add_member(std::move(field));
2017}
