34/**
35 * @brief Base class for standalone diagram elements.
36 *
37 * This is a base cass of any standalone elements such as classes, structs,
38 * concepts, packages and so on participants and so on.
39 */
40class diagram_element : public decorated_element, public source_location {
41public:
42 using id_t = int64_t;
43
44 diagram_element();
45
46 ~diagram_element() override = default;
47
48 /**
49 * @brief Returns diagram element id.
50 *
51 * Each element in the diagram is uniquely identified by id. The id
52 * is currently calculated from the full string representation of the
53 * element, in order to be uniquely identifiable among multiple translation
54 * units.
55 *
56 * @return Elements id.
57 */
58 id_t id() const;
59
60 /**
61 * Set elements id.
62 *
63 * @param id Elements id.
64 */
65 void set_id(id_t id);
66
67 /**
68 * @brief Return elements' diagram alias.
69 *
70 * @todo This is a PlantUML specific method - it shouldn't be here.
71 *
72 * @return PlantUML's diagram element alias.
73 */
74 virtual std::string alias() const;
75
76 /**
77 * Set diagram elements name.
78 *
79 * @param name Elements name.
80 */
81 void set_name(const std::string &name) { name_ = name; }
82
83 /**
84 * Return diagram's name.
85 *
86 * @return Diagram's name.
87 */
88 std::string name() const { return name_; }
89
90 /**
91 * Return the type name of the diagram.
92 *
93 * @return Diagrams type name.
94 */
95 virtual std::string type_name() const { return "__undefined__"; };
96
97 /**
98 * @brief Return the elements fully qualified name.
99 *
100 * This method should be implemented in each subclass, and ensure that
101 * for instance it includes fully qualified namespace, template params, etc.
102 *
103 * @return Full elements name.
104 */
105 virtual std::string full_name(bool /*relative*/) const { return name(); }
106
107 /**
108 * Return all relationships outgoing from this element.
109 *
110 * @return List of relationships.
111 */
112 std::vector<relationship> &relationships();
113
114 /**
115 * Return all relationships outgoing from this element.
116 *
117 * @return List of relationships.
118 */
119 const std::vector<relationship> &relationships() const;
120
121 /**
122 * Add relationships, whose source is this element.
123 *
124 * @param cr Relationship to another diagram element.
125 */
126 void add_relationship(relationship &&cr);
127
128 /**
129 * Add element to the diagram.
130 *
131 * @param e Diagram element.
132 */
133 void append(const decorated_element &e);
134
135 friend bool operator==(const diagram_element &l, const diagram_element &r);
136
137 friend std::ostream &operator<<(
138 std::ostream &out, const diagram_element &rhs);
139
140 /**
141 * Return elements inja JSON context.
142 *
143 * @return Element context.
144 */
145 virtual inja::json context() const;
146
147 /**
148 * Whether this element is nested in another element.
149 *
150 * @return
151 */
152 bool is_nested() const;
153
154 /**
155 * Set element's nested status.
156 *
157 * @param nested
158 */
159 void nested(bool nested);
160
161 /**
162 * Returns the diagrams completion status.
163 *
164 * @return Whether the diagram is complete.
165 */
166 bool complete() const;
167
168 /**
169 * Set the diagrams completion status.
170 *
171 * @param completed
172 */
173 void complete(bool completed);
174
175private:
176 id_t id_{0};
177 std::string name_;
178 std::vector<relationship> relationships_;
179 bool nested_{false};
180 bool complete_{false};
181};
182} // namespace clanguml::common::model
