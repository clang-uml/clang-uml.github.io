280template <typename ElementT>
281bool diagram::add_with_namespace_path(std::unique_ptr<ElementT> &&e)
282{
283 const auto base_name = e->name();
284 const auto full_name = e->full_name(false);
285 const auto element_type = e->type_name();
286
287 LOG_DBG("Adding {}: {}::{}, {}", element_type,
288 e->get_namespace().to_string(), base_name, full_name);
289
290 if (util::contains(base_name, "::"))
291 throw std::runtime_error("Name cannot contain namespace: " + base_name);
292
293 if (util::contains(base_name, "*"))
294 throw std::runtime_error("Name cannot contain *: " + base_name);
295
296 const auto ns = e->get_relative_namespace();
297 auto name = base_name;
298 auto name_with_ns = e->name_and_ns();
299 auto name_and_ns = ns | name;
300 auto &e_ref = *e;
301 auto id = e_ref.id();
302
303 try {
304 if (!contains(e_ref)) {
305 if (add_element(ns, std::move(e)))
306 element_view<ElementT>::add(std::ref(e_ref));
307
308 const auto &el = get_element<ElementT>(name_and_ns).value();
309
310 if ((el.name() != name) || !(el.get_relative_namespace() == ns))
311 throw std::runtime_error(
312 "Invalid element stored in the diagram tree");
313
314 LOG_DBG("Added {} {} ({} - [{}])", element_type, base_name,
315 full_name, id);
316
317 return true;
318 }
319 }
320 catch (const std::runtime_error &e) {
321 LOG_WARN("Cannot add {} {} with id {} due to: {}", element_type, name,
322 id, e.what());
323 return false;
324 }
325
326 LOG_DBG("{} {} ({} - [{}]) already in the model", element_type, base_name,
327 full_name, id);
328
329 return false;
330}

332template <typename ElementT>
333bool diagram::add_with_module_path(
334 const common::model::path &parent_path, std::unique_ptr<ElementT> &&e)
335{
336 const auto element_type = e->type_name();
337
338 // Make sure all parent modules are already packages in the
339 // model
340 for (auto it = parent_path.begin(); it != parent_path.end(); it++) {
341 auto pkg = std::make_unique<common::model::package>(
342 e->using_namespace(), parent_path.type());
343 pkg->set_name(*it);
344 auto ns =
345 common::model::path(parent_path.begin(), it, parent_path.type());
346 // ns.pop_back();
347 pkg->set_namespace(ns);
348 pkg->set_id(common::to_id(pkg->full_name(false)));
349
350 add(ns, std::move(pkg));
351 }
352
353 const auto base_name = e->name();
354 const auto full_name = e->full_name(false);
355 auto &e_ref = *e;
356
357 if (add_element(parent_path, std::move(e))) {
358 element_view<ElementT>::add(std::ref(e_ref));
359 return true;
360 }
361
362 return false;
363}

398template <typename ElementT>
399opt_ref<ElementT> diagram::find(const std::string &name) const
400{
401 for (const auto &element : element_view<ElementT>::view()) {
402 const auto full_name = element.get().full_name(false);
403
404 if (full_name == name) {
405 return {element};
406 }
407 }
408
409 return {};
410}

412template <typename ElementT>
413std::vector<opt_ref<ElementT>> diagram::find(
414 const common::string_or_regex &pattern) const
415{
416 std::vector<opt_ref<ElementT>> result;
417
418 for (const auto &element : element_view<ElementT>::view()) {
419 const auto full_name = element.get().full_name(false);
420
421 if (pattern == full_name) {
422 result.emplace_back(element);
423 }
424 }
425
426 return result;
427}
