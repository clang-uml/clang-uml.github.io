49std::string diagram::to_alias(const clanguml::common::id_t id) const
50{
51 LOG_DBG("Looking for alias for {}", id);
52
53 auto p = find<package>(id);
54 if (p.has_value() && p.value().id() == id)
55 return p.value().alias();
56
57 return {};
58}

60inja::json diagram::context() const
61{
62 inja::json ctx;
63 ctx["name"] = name();
64 ctx["type"] = "package";
65
66 inja::json::array_t elements{};
67
68 for (const auto &p : packages()) {
69 elements.emplace_back(p.get().context());
70 }
71
72 ctx["elements"] = elements;
73
74 return ctx;
75}
