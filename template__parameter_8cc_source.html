27std::string context::to_string() const
28{
29 std::vector<std::string> cv_qualifiers;
30 if (is_const)
31 cv_qualifiers.emplace_back("const");
32 if (is_volatile)
33 cv_qualifiers.emplace_back("volatile");
34
35 auto res = fmt::format("{}", fmt::join(cv_qualifiers, " "));
36
37 if (pr == rpqualifier::kPointer)
38 res += "*";
39 else if (pr == rpqualifier::kLValueReference)
40 res += "&";
41 else if (pr == rpqualifier::kRValueReference)
42 res += "&&";
43
44 if (is_ref_const)
45 res += " const";
46 if (is_ref_volatile)
47 res += " volatile";
48
49 return res;
50}

229int template_parameter::calculate_specialization_match(
230 const template_parameter &base_template_parameter) const
231{
232 int res{0};
233
234 // If the potential base template has a deduction context (e.g. const&),
235 // the specialization must have the same and possibly more
236 if (base_template_parameter.is_specialization()) {
237 if (!deduced_context().empty() &&
238 (base_template_parameter.deduced_context().empty() ||
239 !util::starts_with(deduced_context(),
240 base_template_parameter.deduced_context())))
241 return 0;
242
243 if (!base_template_parameter.deduced_context().empty() &&
244 deduced_context().empty())
245 return 0;
246 }
247
248 if (is_template_parameter() &&
249 base_template_parameter.is_template_parameter() &&
250 template_params().empty() &&
251 base_template_parameter.template_params().empty() &&
252 is_variadic() == is_variadic() &&
253 is_function_template() ==
254 base_template_parameter.is_function_template() &&
255 is_member_pointer() == base_template_parameter.is_member_pointer()) {
256 return 1;
257 }
258
259 auto maybe_base_template_parameter_type = base_template_parameter.type();
260 auto maybe_template_parameter_type = type();
261
262 if (maybe_base_template_parameter_type.has_value() &&
263 maybe_template_parameter_type.has_value() &&
264 !base_template_parameter.is_template_parameter() &&
265 !is_template_parameter()) {
266 if (maybe_base_template_parameter_type.value() !=
267 maybe_template_parameter_type.value())
268 return 0;
269
270 res++;
271 }
272
273 if (base_template_parameter.is_array() && !is_array())
274 return 0;
275
276 if (base_template_parameter.is_function_template() &&
277 !is_function_template())
278 return 0;
279
280 if (base_template_parameter.is_member_pointer() && !is_member_pointer())
281 return 0;
282
283 if (base_template_parameter.is_data_pointer() && !is_data_pointer())
284 return 0;
285
286 if (!base_template_parameter.template_params().empty() &&
287 !template_params().empty() &&
288 is_same_specialization(base_template_parameter)) {
289 auto params_match = calculate_template_params_specialization_match(
290 template_params(), base_template_parameter.template_params());
291
292 if (params_match == 0)
293 return 0;
294
295 res += params_match;
296 }
297 else if ((base_template_parameter.is_template_parameter() ||
298 base_template_parameter.is_template_template_parameter()) &&
299 !is_template_parameter()) {
300 return 1;
301 }
302 else if (base_template_parameter.is_template_parameter() &&
303 base_template_parameter.template_params().empty()) {
304 // If the base is a regular template param, only possible with deduced
305 // context (deduced context already matches if exists)
306 res++;
307
308 if (!deduced_context().empty() &&
309 !base_template_parameter.deduced_context().empty() &&
310 util::starts_with(
311 deduced_context(), base_template_parameter.deduced_context()))
312 res += static_cast<int>(
313 base_template_parameter.deduced_context().size());
314 }
315
316 return res;
317}

335bool operator==(const template_parameter &l, const template_parameter &r)
336{
337 bool res{false};
338
339 if (l.is_template_parameter() != r.is_template_parameter())
340 return res;
341
342 if (l.is_function_template() != r.is_function_template())
343 return res;
344
345 if (l.is_template_parameter()) {
346 // If this is a template parameter (e.g. 'typename T' or 'typename U'
347 // we don't actually care what it is called
348 res = (l.is_variadic() == r.is_variadic()) &&
349 (l.default_value() == r.default_value());
350 }
351 else
352 res = (l.name() == r.name()) && (l.type() == r.type()) &&
353 (l.default_value() == r.default_value());
354
355 return res && (l.template_params_ == r.template_params_);
356}
357
358bool operator!=(const template_parameter &l, const template_parameter &r)
359{
360 return !(l == r);
361}

374std::string template_parameter::to_string(
375 const clanguml::common::model::namespace_ &using_namespace, bool relative,
376 bool skip_qualifiers) const
377{
378 if (is_ellipsis())
379 return "...";
380
381 using clanguml::common::model::namespace_;
382
383 assert(!(type().has_value() && concept_constraint().has_value()));
384
385 if (is_array()) {
386 auto it = template_params_.begin();
387 auto element_type = it->to_string(using_namespace, relative);
388 std::advance(it, 1);
389
390 std::vector<std::string> dimension_args;
391 for (; it != template_params_.end(); it++)
392 dimension_args.push_back(it->to_string(using_namespace, relative));
393
394 return fmt::format(
395 "{}[{}]", element_type, fmt::join(dimension_args, "]["));
396 }
397
398 if (is_function_template()) {
399 auto it = template_params_.begin();
400 auto return_type = it->to_string(using_namespace, relative);
401 std::advance(it, 1);
402
403 std::vector<std::string> function_args;
404 for (; it != template_params_.end(); it++)
405 function_args.push_back(it->to_string(using_namespace, relative));
406
407 return fmt::format(
408 "{}({})", return_type, fmt::join(function_args, ","));
409 }
410
411 if (is_data_pointer()) {
412 assert(template_params().size() == 2);
413
414 std::string unqualified = fmt::format("{} {}::*",
415 template_params().at(0).to_string(using_namespace, relative),
416 template_params().at(1).to_string(using_namespace, relative));
417
418 if (skip_qualifiers)
419 return unqualified;
420
421 return util::join(" ", unqualified, deduced_context_str());
422 }
423
424 if (is_member_pointer()) {
425 assert(template_params().size() > 1);
426
427 auto it = template_params().begin();
428 auto return_type = it->to_string(using_namespace, relative);
429 it++;
430 auto class_type = it->to_string(using_namespace, relative);
431 it++;
432 std::vector<std::string> args;
433
434 for (; it != template_params().end(); it++) {
435 args.push_back(it->to_string(using_namespace, relative));
436 }
437
438 std::string unqualified = fmt::format(
439 "{} ({}::*)({})", return_type, class_type, fmt::join(args, ","));
440 if (skip_qualifiers)
441 return unqualified;
442
443 return util::join(" ", unqualified, deduced_context_str());
444 }
445
446 std::string res;
447 const auto maybe_type = type();
448 if (maybe_type) {
449 if (!relative)
450 res += namespace_{*maybe_type}.to_string();
451 else
452 res += namespace_{*maybe_type}
453 .relative_to(using_namespace)
454 .to_string();
455 }
456
457 const auto &maybe_concept_constraint = concept_constraint();
458
459 if (maybe_concept_constraint) {
460 if (!relative)
461 res += namespace_{maybe_concept_constraint.value()}.to_string();
462 else
463 res += namespace_{maybe_concept_constraint.value()}
464 .relative_to(using_namespace)
465 .to_string();
466 }
467
468 const auto maybe_name = name();
469
470 if (maybe_name) {
471 if ((maybe_type && !maybe_type.value().empty()) ||
472 maybe_concept_constraint)
473 res += " ";
474
475 if (!relative)
476 res += namespace_{*maybe_name}.to_string();
477 else
478 res += namespace_{*maybe_name}
479 .relative_to(using_namespace)
480 .to_string();
481 }
482
483 // Render nested template params
484 if (!template_params_.empty()) {
485 std::vector<std::string> params;
486 params.reserve(template_params_.size());
487 for (const auto &template_param : template_params_) {
488 params.push_back(
489 template_param.to_string(using_namespace, relative));
490 }
491
492 res += fmt::format("<{}>", fmt::join(params, ","));
493 }
494
495 if (!skip_qualifiers)
496 res = util::join(" ", res, deduced_context_str());
497
498 const auto &maybe_default_value = default_value();
499 if (maybe_default_value) {
500 res += "=";
501 res += maybe_default_value.value();
502 }
503
504 return res;
505}

507bool template_parameter::find_nested_relationships(
508 std::vector<std::pair<int64_t, common::model::relationship_t>>
509 &nested_relationships,
510 common::model::relationship_t hint,
511 const std::function<bool(const std::string &full_name)> &should_include)
512 const
513{
514 bool added_aggregation_relationship{false};
515
516 // If this type argument should be included in the relationship
517 // just add it and skip recursion (e.g. this is a user defined type)
518 const auto maybe_type = type();
519
520 if (maybe_type && should_include(maybe_type.value())) {
521 if (is_association())
522 hint = common::model::relationship_t::kAssociation;
523
524 const auto maybe_id = id();
525 if (maybe_id) {
526 nested_relationships.emplace_back(maybe_id.value(), hint);
527 added_aggregation_relationship =
528 (hint == common::model::relationship_t::kAggregation);
529 }
530 }
531 // Otherwise (e.g. this is a std::shared_ptr) and we're actually
532 // interested what is stored inside it
533 else {
534 for (const auto &template_argument : template_params()) {
535
536 const auto maybe_id = template_argument.id();
537 const auto maybe_arg_type = template_argument.type();
538
539 if (maybe_id && maybe_arg_type && should_include(*maybe_arg_type)) {
540
541 if (template_argument.is_association())
542 hint = common::model::relationship_t::kAssociation;
543
544 nested_relationships.emplace_back(maybe_id.value(), hint);
545
546 added_aggregation_relationship =
547 (hint == common::model::relationship_t::kAggregation);
548 }
549 else {
550 added_aggregation_relationship =
551 template_argument.find_nested_relationships(
552 nested_relationships, hint, should_include);
553 }
554 }
555 }
556
557 return added_aggregation_relationship;
558}

654int calculate_template_params_specialization_match(
655 const std::vector<template_parameter> &specialization_params,
656 const std::vector<template_parameter> &template_params)
657{
658 int res{0};
659
660 if (specialization_params.size() != template_params.size() &&
661 !std::any_of(template_params.begin(), template_params.end(),
662 [](const auto &t) { return t.is_variadic(); })) {
663 return 0;
664 }
665
666 if (!specialization_params.empty() && !template_params.empty()) {
667 auto template_index{0U};
668 auto arg_index{0U};
669
670 while (arg_index < specialization_params.size() &&
671 template_index < template_params.size()) {
672 auto match = specialization_params.at(arg_index)
673 .calculate_specialization_match(
674 template_params.at(template_index));
675
676 if (match == 0) {
677 // If any of the matches is 0 - the entire match fails
678 return 0;
679 }
680
681 // Add 1 point if the current specialization param is an argument
682 // as it's a more specific match than 2 template params
683 if (!specialization_params.at(arg_index).is_template_parameter())
684 res++;
685
686 // Add 1 point if the current template param is an argument
687 // as it's a more specific match than 2 template params
688 if (!template_params.at(template_index).is_template_parameter())
689 res++;
690
691 if (!template_params.at(template_index).is_variadic())
692 template_index++;
693
694 res += match;
695
696 arg_index++;
697 }
698
699 if (arg_index == specialization_params.size()) {
700 // Check also backwards to make sure that trailing non-variadic
701 // params match after a variadic parameter
702 template_index = template_params.size() - 1;
703 arg_index = specialization_params.size() - 1;
704
705 while (true) {
706 auto match = specialization_params.at(arg_index)
707 .calculate_specialization_match(
708 template_params.at(template_index));
709 if (match == 0) {
710 return 0;
711 }
712
713 if (arg_index == 0 || template_index == 0)
714 break;
715
716 arg_index--;
717
718 if (!template_params.at(template_index).is_variadic())
719 template_index--;
720 else
721 break;
722 }
723
724 return res;
725 }
726
727 return 0;
728 }
729
730 return 0;
