45/**
46 * @brief Simple optional reference type.
47 *
48 * This class template provides a convenient way around the std::optional
49 * limitation of not allowing references. This is useful for storing
50 * references to diagram elements in various `views`, and writing methods
51 * which return can return an empty value if the diagram does not contain
52 * something.
53 *
54 * This is not an owning type - it will not accept an rvalue - the actual
55 * value must be stored somewhere else as lvalue or some kind of smart pointer.
56 *
57 * @note Probably unsafe - do not use at home.
58 *
59 * @tparam T Type of reference
60 */
61template <typename T> class optional_ref {
62public:
63 using optional_type = T;
64
65 optional_ref() = default;
66
67 optional_ref(T *value)
68 : value_{value}
69 {
70 }
71
72 optional_ref(const T *value)
73 : value_{value}
74 {
75 }
76
77 optional_ref(T &value)
78 : value_{&value}
79 {
80 }
81
82 optional_ref(const T &value)
83 : value_{&value}
84 {
85 }
86
87 optional_ref(optional_ref &right)
88 : value_{right.get()}
89 {
90 }
91
92 template <typename V,
93 typename = std::enable_if<
94 std::is_base_of_v<optional_type, typename V::optional_type> ||
95 std::is_same_v<optional_type, typename V::optional_type>>>
96 optional_ref(const V &t)
97 : value_{t.get()}
98 {
99 }
100
101 template <typename V,
102 typename = std::enable_if<
103 std::is_base_of_v<optional_type, typename V::optional_type> ||
104 std::is_same_v<optional_type, typename V::optional_type>>>
105 optional_ref(V &&t)
106 : value_{t.get()}
107 {
108 t.reset();
109 }
110
111 template <typename V,
112 typename = std::enable_if<std::is_base_of_v<optional_type, V>>>
113 optional_ref(const std::reference_wrapper<V> &t)
114 : value_{&t.get()}
115 {
116 }
117
118 optional_ref &operator=(const optional_ref &right)
119 {
120 if (this == &right)
121 return *this;
122
123 value_ = right.value_;
124 return *this;
125 }
126
127 optional_ref &operator=(optional_ref &&right) noexcept
128 {
129 if (this == &right)
130 return *this;
131
132 value_ = right.value_;
133 right.reset();
134 return *this;
135 }
136
137 bool has_value() const noexcept { return value_ != nullptr; }
138
139 operator bool() const noexcept { return has_value(); }
140
141 const T &value() const
142 {
143 assert(value_ != nullptr);
144 return *value_;
145 }
146
147 T &value()
148 {
149 assert(value_ != nullptr);
150 return *value_;
151 }
152
153 T &operator*()
154 {
155 assert(value_ != nullptr);
156 return *value_;
157 }
158
159 const T &operator*() const
160 {
161 assert(value_ != nullptr);
162 return *value_;
163 }
164
165 void reset() { value_ = nullptr; }
166
167 T *get() const { return value_; }
168
169private:
170 T *value_{nullptr};
171};

181/**
182 * @brief Wrapper around std::regex, which contains original pattern
183 */
184struct regex {
185 /**
186 * @brief Constructor
187 *
188 * @param r Parsed regular expression
189 * @param p Raw regular expression pattern used for regenerating config and
190 * debugging
191 */
192 regex(std::regex r, std::string p)
193 : regexp{std::move(r)}
194 , pattern{std::move(p)}
195 {
196 }
197
198 /**
199 * @brief Regular expression match operator
200 * @param v Value to match against internal std::regex
201 * @return True, if the argument matches the regular expression
202 */
203 [[nodiscard]] bool operator%=(const std::string &v) const
204 {
205 return std::regex_match(v, regexp);
206 }
207
208 std::regex regexp; /*!< Parsed regular expression */
209 std::string pattern; /*!< Original regular expression pattern */
210};

212/**
213 * @brief Convenience class for configuration options with regex support
214 *
215 * This template class provides a convenient way of handling configuraiton
216 * options, which can be either some basic type like std::string or regex.
217 *
218 * @tparam T Type of alternative to regex (e.g. std::string)
219 */
220template <typename T> struct or_regex {
221 or_regex() = default;
222
223 /**
224 * @brief Constructor from alternative type
225 */
226 or_regex(T v)
227 : value_{std::move(v)}
228 {
229 }
230
231 /**
232 * @brief Constructor from regex
233 *
234 * @param r Parsed regular expression
235 * @param p Raw regular expression pattern used for regenerating config and
236 * debugging
237 */
238 or_regex(std::regex r, std::string p)
239 : value_{regex{std::move(r), std::move(p)}}
240 {
241 }
242
243 or_regex &operator=(const T &v)
244 {
245 value_ = v;
246 return *this;
247 }
248
249 or_regex &operator=(const regex &v)
250 {
251 value_ = v;
252 return *this;
253 }
254
255 [[nodiscard]] bool operator==(const T &v) const
256 {
257 if (std::holds_alternative<regex>(value_))
258 return std::regex_match(v, std::get<regex>(value_).regexp);
259
260 return std::get<T>(value_) == v;
261 }
262
263 template <typename Ret> std::optional<Ret> get() const
264 {
265 if (!std::holds_alternative<Ret>(value_))
266 return std::nullopt;
267
268 return std::get<Ret>(value_);
269 }
270
271 std::string to_string() const
272 {
273 if (std::holds_alternative<regex>(value_))
274 return std::get<regex>(value_).pattern;
275
276 return clanguml::common::to_string(std::get<T>(value_));
277 }
278
279 const std::variant<T, regex> &value() const { return value_; }
280
281private:
282 std::variant<T, regex> value_;
283};
