86std::string class_::full_name(bool relative) const
87{
88 using namespace clanguml::util;
89 using clanguml::common::model::namespace_;
90
91 std::ostringstream ostr;
92
93 ostr << name_and_ns();
94
95 render_template_params(ostr, using_namespace(), relative);
96
97 std::string res;
98
99 if (relative)
100 res = using_namespace().relative(ostr.str());
101 else
102 res = ostr.str();
103
104 if (res.empty())
105 return "<<anonymous>>";
106
107 return res;
108}

139std::optional<std::string> class_::doxygen_link() const
140{
141 const auto *type = is_struct() ? "struct" : "class";
142
143 auto name = name_and_ns();
144 util::replace_all(name, "_", "__");
145 util::replace_all(name, "::", "_1_1");
146 util::replace_all(name, "##", "_1_1"); // nested classes
147 return fmt::format("{}{}.html", type, name);
148}
