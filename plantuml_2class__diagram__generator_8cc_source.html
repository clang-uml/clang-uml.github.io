122void generator::generate(const class_ &c, std::ostream &ostr) const
123{
124 namespace plantuml_common = clanguml::common::generators::plantuml;
125
126 std::string class_type{"class"};
127 if (c.is_abstract())
128 class_type = "abstract";
129
130 ostr << class_type << " " << c.alias();
131
132 if (c.is_union())
133 ostr << " "
134 << "<<union>>";
135
136 if (config().generate_links) {
137 common_generator<diagram_config, diagram_model>::generate_link(ostr, c);
138 }
139
140 if (!c.style().empty())
141 ostr << " " << c.style();
142
143 ostr << " {" << '\n';
144
145 //
146 // Process methods
147 //
148 if (config().group_methods()) {
149 generate_methods(group_methods(c.methods()), ostr);
150 }
151 else {
152 generate_methods(c.methods(), ostr);
153 }
154
155 //
156 // Process relationships - here only generate the set of
157 // rendered_relationships we'll generate them in a seperate method
158 //
159 std::set<std::string> rendered_relations;
160
161 std::stringstream all_relations_str;
162 for (const auto &r : c.relationships()) {
163 if (!model().should_include(r.type()))
164 continue;
165
166 try {
167 generate_relationship(r, rendered_relations);
168 }
169 catch (error::uml_alias_missing &e) {
170 LOG_DBG("Skipping {} relation from {} to {} due "
171 "to: {}",
172 plantuml_common::to_plantuml(r.type(), r.style()),
173 c.full_name(), r.destination(), e.what());
174 }
175 }
176
177 //
178 // Process members
179 //
180 std::vector<clanguml::class_diagram::model::class_member> members{
181 c.members()};
182
183 sort_class_elements(members);
184
185 if (config().group_methods())
186 ostr << "__\n";
187
188 for (const auto &m : members) {
189 if (!model().should_include(m))
190 continue;
191
192 if (!config().include_relations_also_as_members() &&
193 rendered_relations.find(m.name()) != rendered_relations.end())
194 continue;
195
196 generate_member(m, ostr);
197
198 ostr << '\n';
199 }
200
201 ostr << "}" << '\n';
202
203 generate_notes(ostr, c);
204
205 for (const auto &member : c.members())
206 generate_member_notes(ostr, member, c.alias());
207
208 for (const auto &method : c.methods())
209 generate_member_notes(ostr, method, c.alias());
210}

244generator::method_groups_t generator::group_methods(
245 const std::vector<class_method> &methods) const
246{
247 std::map<std::string, std::vector<class_method>> result;
248
249 // First get rid of methods which don't pass the filters
250 std::vector<class_method> filtered_methods;
251 std::copy_if(methods.cbegin(), methods.cend(),
252 std::back_inserter(filtered_methods),
253 [this](auto &m) { return model().should_include(m); });
254
255 for (const auto &g : method_groups_) {
256 result[g] = {};
257 }
258
259 for (const auto &m : filtered_methods) {
260 if (m.is_constructor() || m.is_destructor()) {
261 result["constructors"].push_back(m);
262 }
263 else if (m.is_copy_assignment() || m.is_move_assignment()) {
264 result["assignment"].push_back(m);
265 }
266 else if (m.is_operator()) {
267 result["operators"].push_back(m);
268 }
269 else {
270 result["other"].push_back(m);
271 }
272 }
273
274 return result;
275}

277void generator::generate_method(
278 const class_diagram::model::class_method &m, std::ostream &ostr) const
279{
280 namespace plantuml_common = clanguml::common::generators::plantuml;
281 const auto &uns = config().using_namespace();
282
283 constexpr auto kAbbreviatedMethodArgumentsLength{15};
284
285 print_debug(m, ostr);
286
287 if (m.is_pure_virtual())
288 ostr << "{abstract} ";
289
290 if (m.is_static())
291 ostr << "{static} ";
292
293 std::string type{uns.relative(config().simplify_template_type(m.type()))};
294
295 ostr << plantuml_common::to_plantuml(m.access()) << m.name();
296
297 if (!m.template_params().empty()) {
298 m.render_template_params(ostr, config().using_namespace(), false);
299 }
300
301 ostr << "(";
302 if (config().generate_method_arguments() !=
303 config::method_arguments::none) {
304 std::vector<std::string> params;
305 std::transform(m.parameters().cbegin(), m.parameters().cend(),
306 std::back_inserter(params), [this](const auto &mp) {
307 return config().simplify_template_type(
308 mp.to_string(config().using_namespace()));
309 });
310 auto args_string = fmt::format("{}", fmt::join(params, ", "));
311 if (config().generate_method_arguments() ==
312 config::method_arguments::abbreviated) {
313 args_string = clanguml::util::abbreviate(
314 args_string, kAbbreviatedMethodArgumentsLength);
315 }
316 ostr << args_string;
317 }
318 ostr << ")";
319
320 if (m.is_constexpr())
321 ostr << " constexpr";
322 else if (m.is_consteval())
323 ostr << " consteval";
324
325 if (m.is_const())
326 ostr << " const";
327
328 if (m.is_noexcept())
329 ostr << " noexcept";
330
331 assert(!(m.is_pure_virtual() && m.is_defaulted()));
332
333 if (m.is_pure_virtual())
334 ostr << " = 0";
335
336 if (m.is_defaulted())
337 ostr << " = default";
338 else if (m.is_deleted())
339 ostr << " = deleted";
340
341 ostr << " : " << type;
342
343 if (config().generate_links) {
344 generate_link(ostr, m);
345 }
346}

439void generator::generate_relationship(
440 const relationship &r, std::set<std::string> &rendered_relations) const
441{
442 namespace plantuml_common = clanguml::common::generators::plantuml;
443
444 LOG_DBG("Processing relationship {}",
445 plantuml_common::to_plantuml(r.type(), r.style()));
446
447 std::string destination;
448
449 auto target_element = model().get(r.destination());
450 if (!target_element.has_value())
451 throw error::uml_alias_missing{fmt::format(
452 "Missing element in the model for ID: {}", r.destination())};
453
454 destination = target_element.value().full_name(false);
455
456 if (util::starts_with(destination, std::string{"::"}))
457 destination = destination.substr(2, destination.size());
458
459 std::string puml_relation;
460 if (!r.multiplicity_source().empty())
461 puml_relation += "\"" + r.multiplicity_source() + "\" ";
462
463 puml_relation += plantuml_common::to_plantuml(r.type(), r.style());
464
465 if (!r.multiplicity_destination().empty())
466 puml_relation += " \"" + r.multiplicity_destination() + "\"";
467
468 if (!r.label().empty()) {
469 rendered_relations.emplace(r.label());
470 }
471}

473void generator::generate_relationships(
474 const class_ &c, std::ostream &ostr) const
475{
476 namespace plantuml_common = clanguml::common::generators::plantuml;
477
478 //
479 // Process relationships
480 //
481 std::set<std::string> rendered_relations;
482
483 std::stringstream all_relations_str;
484 std::set<std::string> unique_relations;
485
486 for (const auto &r : c.relationships()) {
487 if (!model().should_include(r.type()))
488 continue;
489
490 LOG_DBG("== Processing relationship {}",
491 plantuml_common::to_plantuml(r.type(), r.style()));
492
493 std::stringstream relstr;
494 clanguml::common::id_t destination{0};
495 try {
496 destination = r.destination();
497
498 std::string puml_relation;
499 if (!r.multiplicity_source().empty())
500 puml_relation += "\"" + r.multiplicity_source() + "\" ";
501
502 puml_relation += plantuml_common::to_plantuml(r.type(), r.style());
503
504 if (!r.multiplicity_destination().empty())
505 puml_relation += " \"" + r.multiplicity_destination() + "\"";
506
507 std::string target_alias;
508 try {
509 target_alias = model().to_alias(destination);
510 }
511 catch (...) {
512 LOG_DBG("Failed to find alias to {}", destination);
513 continue;
514 }
515
516 if (m_generated_aliases.find(target_alias) ==
517 m_generated_aliases.end())
518 continue;
519
520 relstr << c.alias() << " " << puml_relation << " " << target_alias;
521
522 if (!r.label().empty()) {
523 relstr << " : " << plantuml_common::to_plantuml(r.access())
524 << r.label();
525 rendered_relations.emplace(r.label());
526 }
527
528 if (unique_relations.count(relstr.str()) == 0) {
529 unique_relations.emplace(relstr.str());
530
531 relstr << '\n';
532
533 LOG_DBG("=== Adding relation {}", relstr.str());
534
535 all_relations_str << relstr.str();
536 }
537 }
538 catch (error::uml_alias_missing &e) {
539 LOG_DBG("=== Skipping {} relation from {} to {} due "
540 "to: {}",
541 plantuml_common::to_plantuml(r.type(), r.style()),
542 c.full_name(), destination, e.what());
543 }
544 }
545
546 if (model().should_include(relationship_t::kExtension)) {
547 for (const auto &b : c.parents()) {
548 std::stringstream relstr;
549 try {
550 auto target_alias = model().to_alias(b.id());
551
552 if (m_generated_aliases.find(target_alias) ==
553 m_generated_aliases.end())
554 continue;
555
556 relstr << target_alias << " <|-- " << c.alias() << '\n';
557 all_relations_str << relstr.str();
558 }
559 catch (error::uml_alias_missing &e) {
560 LOG_DBG("=== Skipping inheritance relation from {} to {} due "
561 "to: {}",
562 b.name(), c.name(), e.what());
563 }
564 }
565 }
566
567 ostr << all_relations_str.str();
568}

707void generator::generate(const package &p, std::ostream &ostr) const
708{
709 const auto &uns = config().using_namespace();
710
711 if (config().generate_packages()) {
712 LOG_DBG("Generating package {}", p.name());
713
714 // Don't generate packages from namespaces filtered out by
715 // using_namespace
716 if (!uns.starts_with({p.full_name(false)})) {
717 print_debug(p, ostr);
718 ostr << "package [" << p.name() << "] ";
719 ostr << "as " << p.alias();
720
721 if (p.is_deprecated())
722 ostr << " <<deprecated>>";
723
724 if (!p.style().empty())
725 ostr << " " << p.style();
726
727 ostr << " {" << '\n';
728 }
729 }
730
731 for (const auto &subpackage : p) {
732 if (dynamic_cast<package *>(subpackage.get()) != nullptr) {
733 // TODO: add option - generate_empty_packages
734 const auto &sp = dynamic_cast<package &>(*subpackage);
735 if (!sp.is_empty()) {
736 together_group_stack_.enter();
737
738 generate(sp, ostr);
739
740 together_group_stack_.leave();
741 }
742 }
743 else if (auto *cls = dynamic_cast<class_ *>(subpackage.get()); cls) {
744 if (model().should_include(*subpackage)) {
745 auto together_group =
746 config().get_together_group(cls->full_name(false));
747 if (together_group) {
748 together_group_stack_.group_together(
749 together_group.value(), cls);
750 }
751 else {
752 generate_alias(*cls, ostr);
753 generate(*cls, ostr);
754 }
755 }
756 }
757 else if (auto *enm = dynamic_cast<enum_ *>(subpackage.get()); enm) {
758 if (model().should_include(*subpackage)) {
759 auto together_group =
760 config().get_together_group(subpackage->full_name(false));
761 if (together_group) {
762 together_group_stack_.group_together(
763 together_group.value(), enm);
764 }
765 else {
766 generate_alias(*enm, ostr);
767 generate(*enm, ostr);
768 }
769 }
770 }
771 else if (auto *cpt = dynamic_cast<concept_ *>(subpackage.get()); cpt) {
772 if (model().should_include(*subpackage)) {
773 auto together_group =
774 config().get_together_group(cpt->full_name(false));
775 if (together_group) {
776 together_group_stack_.group_together(
777 together_group.value(), cpt);
778 }
779 else {
780 generate_alias(*cpt, ostr);
781 generate(*cpt, ostr);
782 }
783 }
784 }
785 }
786
787 if (config().generate_packages()) {
788 // Now generate any diagram elements which are in together
789 // groups
790 for (const auto &[group_name, group_elements] :
791 together_group_stack_.get_current_groups()) {
792 ostr << "together {\n";
793
794 for (auto *e : group_elements) {
795 if (auto *cls = dynamic_cast<class_ *>(e); cls) {
796 generate_alias(*cls, ostr);
797 generate(*cls, ostr);
798 }
799 if (auto *enm = dynamic_cast<enum_ *>(e); enm) {
800 generate_alias(*enm, ostr);
801 generate(*enm, ostr);
802 }
803 if (auto *cpt = dynamic_cast<concept_ *>(e); cpt) {
804 generate_alias(*cpt, ostr);
805 generate(*cpt, ostr);
806 }
807 }
808
809 ostr << "}\n";
810 }
811
812 // Don't generate packages from namespaces filtered out by
813 // using_namespace
814 if (!uns.starts_with({p.full_name(false)})) {
815 ostr << "}" << '\n';
816 generate_notes(ostr, p);
817 }
818 }
819}

821void generator::generate_relationships(
822 const package &p, std::ostream &ostr) const
823{
824 for (const auto &subpackage : p) {
825 if (dynamic_cast<package *>(subpackage.get()) != nullptr) {
826 // TODO: add option - generate_empty_packages, currently
827 // packages which do not contain anything but other
828 // packages are skipped
829 const auto &sp = dynamic_cast<package &>(*subpackage);
830 if (!sp.is_empty() &&
831 !sp.all_of([this](const common::model::element &e) {
832 return !model().should_include(e);
833 }))
834 generate_relationships(sp, ostr);
835 }
836 else if (dynamic_cast<class_ *>(subpackage.get()) != nullptr) {
837 if (model().should_include(*subpackage)) {
838 generate_relationships(
839 dynamic_cast<class_ &>(*subpackage), ostr);
840 }
841 }
842 else if (dynamic_cast<enum_ *>(subpackage.get()) != nullptr) {
843 if (model().should_include(*subpackage)) {
844 generate_relationships(
845 dynamic_cast<enum_ &>(*subpackage), ostr);
846 }
847 }
848 else if (dynamic_cast<concept_ *>(subpackage.get()) != nullptr) {
849 if (model().should_include(*subpackage)) {
850 generate_relationships(
851 dynamic_cast<concept_ &>(*subpackage), ostr);
852 }
853 }
854 }

868void generator::generate_top_level_elements(std::ostream &ostr) const
869{
870 for (const auto &p : model()) {
871 if (auto *pkg = dynamic_cast<package *>(p.get()); pkg) {
872 if (!pkg->is_empty() &&
873 !pkg->all_of([this](const common::model::element &e) {
874 return !model().should_include(e);
875 }))
876 generate(*pkg, ostr);
877 }
878 else if (auto *cls = dynamic_cast<class_ *>(p.get()); cls) {
879 if (model().should_include(*cls)) {
880 auto together_group =
881 config().get_together_group(cls->full_name(false));
882 if (together_group) {
883 together_group_stack_.group_together(
884 together_group.value(), cls);
885 }
886 else {
887 generate_alias(*cls, ostr);
888 generate(*cls, ostr);
889 }
890 }
891 }
892 else if (auto *enm = dynamic_cast<enum_ *>(p.get()); enm) {
893 if (model().should_include(*enm)) {
894 auto together_group =
895 config().get_together_group(enm->full_name(false));
896 if (together_group) {
897 together_group_stack_.group_together(
898 together_group.value(), enm);
899 }
900 else {
901 generate_alias(*enm, ostr);
902 generate(*enm, ostr);
903 }
904 }
905 }
906 else if (auto *cpt = dynamic_cast<concept_ *>(p.get()); cpt) {
907 if (model().should_include(*cpt)) {
908 auto together_group =
909 config().get_together_group(cpt->full_name(false));
910 if (together_group) {
911 together_group_stack_.group_together(
912 together_group.value(), cpt);
913 }
914 else {
915 generate_alias(*cpt, ostr);
916 generate(*cpt, ostr);
917 }
918 }
919 }
920 }
