25std::unique_ptr<compilation_database>
26compilation_database::auto_detect_from_directory(
27 const clanguml::config::config &cfg)
28{
29 std::string error_message;
30 auto res = clang::tooling::CompilationDatabase::autoDetectFromDirectory(
31 cfg.compilation_database_dir(), error_message);
32
33 if (!error_message.empty())
34 throw error::compilation_database_error(error_message);
35
36 return std::make_unique<compilation_database>(std::move(res), cfg);
37}

82std::string compilation_database::guess_language_from_filename(
83 const std::string &filename) const
84{

91void compilation_database::adjust_compilation_database(
92 std::vector<clang::tooling::CompileCommand> &commands) const
93{
94#if !defined(_WIN32)
95 if (config().query_driver && !config().query_driver().empty()) {
96 for (auto &compile_command : commands) {
97 auto argv0 = config().query_driver() == "."
98 ? compile_command.CommandLine.at(0)
99 : config().query_driver();
100
101 util::query_driver_output_extractor extractor{
102 argv0, guess_language_from_filename(compile_command.Filename)};
103
104 extractor.execute();
105
106 std::vector<std::string> system_header_args;
107 for (const auto &path : extractor.system_include_paths()) {
108 system_header_args.emplace_back("-isystem");
109 system_header_args.emplace_back(path);
110 }
111
112 compile_command.CommandLine.insert(
113 compile_command.CommandLine.begin() + 1,
114 system_header_args.begin(), system_header_args.end());
115
116 if (!extractor.target().empty()) {
117 compile_command.CommandLine.insert(
118 compile_command.CommandLine.begin() + 1,
119 fmt::format("--target={}", extractor.target()));
120 }
121 }
122 }
123#endif
124
125 if (config().add_compile_flags && !config().add_compile_flags().empty()) {
126 for (auto &compile_command : commands) {
127 compile_command.CommandLine.insert(
128 // Add flags after argv[0]
129 compile_command.CommandLine.begin() + 1,
130 config().add_compile_flags().begin(),
131 config().add_compile_flags().end());
132 }
133 }
134
135 if (config().remove_compile_flags &&
136 !config().remove_compile_flags().empty()) {
137 for (auto &compile_command : commands) {
138 for (const auto &flag : config().remove_compile_flags()) {
139 util::erase_if(compile_command.CommandLine,
140 [&flag](const auto &arg) { return flag == arg; });
141 }
142 }
143 }
144}
