115std::string function::message_name(message_render_mode mode) const
116{
117 constexpr auto kAbbreviatedMethodArgumentsLength{15};
118
119 if (mode == message_render_mode::no_arguments) {
120 return fmt::format("{}(){}", name(), is_const() ? " const" : "");
121 }
122 if (mode == message_render_mode::abbreviated) {
123 return fmt::format("{}({}){}", name(),
124 clanguml::util::abbreviate(

126 kAbbreviatedMethodArgumentsLength),
127 is_const() ? " const" : "");
128 }
129
130 return fmt::format("{}({}){}", name(), fmt::join(parameters_, ","),
131 is_const() ? " const" : "");

262std::string function_template::message_name(message_render_mode mode) const
263{
264 constexpr auto kAbbreviatedMethodArgumentsLength{15};
265
266 std::ostringstream s;
267 render_template_params(s, using_namespace(), true);
268 const std::string template_params = s.str();
269
270 if (mode == message_render_mode::no_arguments) {
271 return fmt::format(
272 "{}{}(){}", name(), template_params, is_const() ? " const" : "");
273 }
274 if (mode == message_render_mode::abbreviated) {
275 return fmt::format("{}({}){}", name(),
276 clanguml::util::abbreviate(
277 fmt::format("{}", fmt::join(parameters(), ",")),
278 kAbbreviatedMethodArgumentsLength),
279 is_const() ? " const" : "");
280 }
281
282 return fmt::format("{}{}({}){}", name(), template_params,
283 fmt::join(parameters(), ","), is_const() ? " const" : "");
284}
