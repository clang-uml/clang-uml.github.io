565template <typename T>
566void translation_unit_visitor::process_record_parent_by_type(eid_t parent_id,
567 class_ &c, namespace_ parent_ns, const clang::RecordDecl *decl)
568{
569 // Here we have 2 options, either:
570 // - the parent is a regular C++ class/struct
571 // - the parent is a class template declaration/specialization
572 auto parent_class = diagram().find<T>(parent_id);
573
574 c.set_namespace(parent_ns);
575 const auto cls_name = decl->getNameAsString();
576 if (cls_name.empty()) {
577 // Nested structs can be anonymous
578 if (anonymous_struct_relationships_.count(decl->getID()) > 0) {
579 const auto &[label, hint, access, destination_multiplicity] =
580 anonymous_struct_relationships_[decl->getID()];
581
582 c.set_name(parent_class.value().name() + "##" +
583 fmt::format("({})", label));
584
585 std::string destination_multiplicity_str{};
586 if (destination_multiplicity.has_value()) {
587 destination_multiplicity_str =
588 std::to_string(*destination_multiplicity);
589 }
590
591 parent_class.value().add_relationship(
592 {hint, common::to_id(c.full_name(false)), access, label, "",
593 destination_multiplicity_str});
594 }
595 else
596 c.set_name(parent_class.value().name() + "##" +
597 fmt::format("(anonymous_{})", std::to_string(decl->getID())));
598 }
599 else {
600 c.set_name(
601 parent_class.value().name() + "##" + decl->getNameAsString());
602 }
603
604 c.set_id(common::to_id(c.full_name(false)));
605
606 if (!(decl->getNameAsString().empty())) {
607 // Don't add anonymous structs as contained in the class
608 // as they are already added as aggregations
609 c.add_relationship({relationship_t::kContainment, parent_id});
610 }
611
612 c.nested(true);
613}
