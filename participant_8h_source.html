31/**
32 * @brief Base class for various types of sequence diagram participants
33 */
34struct participant : public common::model::element,
35 public common::model::stylable_element {
36 using common::model::element::element;
37
38 /**
39 * @brief Enum representing stereotype of a participant
40 */
41 enum class stereotype_t {
42 participant = 0,
43 actor,
44 boundary,
45 control,
46 entity,
47 database,
48 collections,
49 queue
50 };
51
52 participant(const participant &) = delete;
53 participant(participant &&) noexcept = delete;
54 participant &operator=(const participant &) = delete;
55 participant &operator=(participant &&) = delete;
56
57 /**
58 * Get the type name of the diagram element.
59 *
60 * @return Type name of the diagram element.
61 */
62 std::string type_name() const override { return "participant"; }
63
64 /**
65 * @brief Create a string representation of the participant
66 *
67 * @return Participant representation as string
68 */
69 virtual std::string to_string() const;
70
71 stereotype_t stereotype_{stereotype_t::participant};
