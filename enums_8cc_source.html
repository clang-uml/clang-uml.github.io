154diagram_t from_string(const std::string &s)
155{
156 if (s == "class")
157 return diagram_t::kClass;
158 if (s == "sequence")
159 return diagram_t::kSequence;
160 if (s == "include")
161 return diagram_t::kInclude;
162 if (s == "package")
163 return diagram_t::kPackage;
164
165 throw std::runtime_error{"Invalid diagram type: " + s};
166}
