30void generator::generate_relationships(
31 const source_file &f, std::ostream &ostr) const
32{
33 if (!util::contains(m_generated_aliases, f.alias()))
34 return;
35
36 LOG_DBG("Generating relationships for file {}", f.full_name(true));
37
38 namespace plantuml_common = clanguml::common::generators::plantuml;
39
40 if (f.type() == common::model::source_file_t::kDirectory) {
41 util::for_each(f, [this, &ostr](const auto &file) {
42 generate_relationships(
43 dynamic_cast<const source_file &>(*file), ostr);
44 });
45 }
46 else {
47 util::for_each_if(
48 f.relationships(),
49 [this](const auto &r) {
50 return m_model.should_include(r.type()) &&
51 util::contains(m_generated_aliases,
52 m_model.get(r.destination()).value().alias());
53 },
54 [&f, &ostr, this](const auto &r) {
55 ostr << f.alias() << " "
56 << plantuml_common::to_plantuml(r.type(), r.style()) << " "
57 << m_model.get(r.destination()).value().alias() << '\n';
58 });
59 }
60}

62void generator::generate(const source_file &f, std::ostream &ostr) const
63{
64
65 if (f.type() == common::model::source_file_t::kDirectory) {
66 LOG_DBG("Generating directory {}", f.name());
67
68 ostr << "folder \"" << f.name();
69 ostr << "\" as " << f.alias();
70 ostr << " {\n";
71
72 util::for_each(f, [this, &ostr](const auto &file) {
73 generate(dynamic_cast<const source_file &>(*file), ostr);
74 });
75
76 ostr << "}" << '\n';
77
78 m_generated_aliases.emplace(f.alias());
79 }
80 else {
81 LOG_DBG("Generating file {}", f.name());
82
83 if (m_model.should_include(f)) {
84 ostr << "file \"" << f.name() << "\" as " << f.alias();
85
86 if (m_config.generate_links) {
87 generate_link(ostr, f);
88 }
89
90 ostr << '\n';
91
92 m_generated_aliases.emplace(f.alias());
93 }
94 }
95}

97void generator::generate(std::ostream &ostr) const
98{
99 update_context();
100
101 ostr << "@startuml" << '\n';
102
103 if (m_config.puml)
104 generate_plantuml_directives(ostr, m_config.puml().before);
105
106 // Generate files and folders
107 util::for_each_if(
108 m_model, [](const auto & /*f*/) { return true; },
109 [this, &ostr](const auto &f) {
110 generate(dynamic_cast<source_file &>(*f), ostr);
111 });
112
113 // Process file include relationships
114 util::for_each(m_model, [this, &ostr](const auto &f) {
115 generate_relationships(dynamic_cast<source_file &>(*f), ostr);
116 });
117
118 generate_config_layout_hints(ostr);
119
120 if (m_config.puml)
121 generate_plantuml_directives(ostr, m_config.puml().after);
122
123 generate_metadata(ostr);
124
125 ostr << "@enduml" << '\n';
126}
