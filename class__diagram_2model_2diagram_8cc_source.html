68 if (res.has_value())

119void diagram::get_parents(
120 clanguml::common::reference_set<class_> &parents) const
121{
122 bool found_new{false};
123 for (const auto &parent : parents) {
124 for (const auto &pp : parent.get().parents()) {
125 auto p = find<class_>(pp.id());
126
127 if (p.has_value()) {
128 auto [it, found] = parents.emplace(std::ref(p.value()));
129 if (found)
130 found_new = true;
131 }
132 }
133 }
134
135 if (found_new) {
136 get_parents(parents);
137 }
138}

140bool diagram::has_element(
141 clanguml::common::model::diagram_element::id_t id) const
142{
143 const auto has_class = std::any_of(classes().begin(), classes().end(),
144 [id](const auto &c) { return c.get().id() == id; });
145
146 if (has_class)
147 return true;
148
149 const auto has_concept = std::any_of(classes().begin(), classes().end(),
150 [id](const auto &c) { return c.get().id() == id; });
151
152 if (has_concept)
153 return true;
154
155 return std::any_of(enums().begin(), enums().end(),
156 [id](const auto &c) { return c.get().id() == id; });
157}

159std::string diagram::to_alias(
160 clanguml::common::model::diagram_element::id_t id) const
161{
162 LOG_DBG("Looking for alias for {}", id);
163
164 for (const auto &c : classes()) {
165 if (c.get().id() == id) {
166 return c.get().alias();
167 }
168 }
169
170 for (const auto &e : enums()) {
171 if (e.get().id() == id)
172 return e.get().alias();
173 }
174
175 for (const auto &c : concepts()) {
176 if (c.get().id() == id)
177 return c.get().alias();
178 }
179
180 throw error::uml_alias_missing(fmt::format("Missing alias for {}", id));
181}

183inja::json diagram::context() const
184{
185 inja::json ctx;
186 ctx["name"] = name();
187 ctx["type"] = "class";
188
189 inja::json::array_t elements{};
190
191 // Add classes
192 for (const auto &c : classes()) {
193 elements.emplace_back(c.get().context());
194 }
195
196 // Add enums
197 for (const auto &e : enums()) {
198 elements.emplace_back(e.get().context());
199 }
200
201 // Add enums
202 for (const auto &c : concepts()) {
203 elements.emplace_back(c.get().context());
204 }
205
206 ctx["elements"] = elements;
207
208 return ctx;
209}
