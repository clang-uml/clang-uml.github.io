43void diagram::add_file(std::unique_ptr<common::model::source_file> &&f)
44{
45 // Don't add the same file more than once
46 if (find<source_file>(f->id()))
47 return;
48
49 LOG_DBG("Adding source file: {}, {}", f->name(), f->fs_path().string());
50
51 auto &ff = *f;
52
53 assert(!ff.name().empty());
54 assert(ff.id() != 0);
55
56 element_view<source_file>::add(ff);
57
58 auto p = ff.path();
59
60 if (!f->path().is_empty()) {
61 // If the parent path is not empty, ensure relative parent directories
62 // of this source_file are in the diagram
63 common::model::filesystem_path parent_path_so_far{
64 common::model::path_type::kFilesystem};
65 for (const auto &directory : f->path()) {
66 auto source_file_path = parent_path_so_far | directory;
67 if (parent_path_so_far.is_empty())
68 source_file_path = {directory};
69
70 auto dir = std::make_unique<common::model::source_file>(
71 std::filesystem::path{source_file_path.to_string()});
72 dir->set_type(common::model::source_file_t::kDirectory);
73
74 assert(!dir->name().empty());
75
76 if (!get_element(source_file_path).has_value()) {
77 add_file(std::move(dir));
78
79 LOG_DBG("Added directory '{}' at path '{}'", directory,
80 parent_path_so_far.to_string());
81 }
82
83 parent_path_so_far.append(directory);
84 }
85 }
86
87 assert(p.type() == common::model::path_type::kFilesystem);
88
89 add_element(p, std::move(f));
90}

92std::string diagram::to_alias(const std::string &full_name) const
93{
94 LOG_DBG("Looking for alias for {}", full_name);
95
96 auto path = common::model::filesystem_path{full_name};
97
98 if (path.is_empty())
99 throw error::uml_alias_missing(
100 fmt::format("Missing alias for '{}'", path.to_string()));
101
102 auto source_file = get_element<common::model::source_file>(path);
103
104 if (!source_file)
105 throw error::uml_alias_missing(
106 fmt::format("Missing alias for '{}'", path.to_string()));
107
108 return source_file.value().alias();
109}

117common::optional_ref<clanguml::common::model::diagram_element>
118diagram::get_with_namespace(
119 const std::string &name, const common::model::namespace_ &ns) const
120{
121 // Convert to preferred OS path
122 std::filesystem::path namePath{name};
123 auto namePreferred = namePath.make_preferred().string();
124
125 auto element_opt = get(namePreferred);
126
127 if (!element_opt) {
128 // If no element matches, try to prepend the 'using_namespace'
129 // value to the element and search again
130 auto fully_qualified_name = ns | namePreferred;
131 element_opt = get(fully_qualified_name.to_string());
132 }
133
134 return element_opt;
135}

137inja::json diagram::context() const
138{
139 inja::json ctx;
140 ctx["name"] = name();
141 ctx["type"] = "include";
142
143 inja::json::array_t elements{};
144
145 // Add files and directories
146 for (const auto &f : files()) {
147 elements.emplace_back(f.get().context());
148 }
149
150 ctx["elements"] = elements;
151
152 return ctx;
153}
