template<typename DiagramT, typename ElementT, typename ConfigEntryT = std::string, typename MatchOverrideT = common::model::element>
struct clanguml::common::model::edge_traversal_filter< DiagramT, ElementT, ConfigEntryT, MatchOverrideT >

Common template for filters involving traversing relationship graph.

This class template provides a common implementation of a diagram relationship graph traversal. It is used for filters, which need to check for instance, whether an element is in some kind of relationship with other element.
