24{
25 if (std::holds_alternative<std::string>(m.value())) {
26 out << std::get<std::string>(m.value());
27 }
28 else {
29 out << YAML::Key << "r" << YAML::Value
30 << std::get<regex>(m.value()).pattern;
31 }

85YAML::Emitter &operator<<(YAML::Emitter &out, const filter &f)
86{
87 out << YAML::BeginMap;
88 if (!f.namespaces.empty())
89 out << YAML::Key << "namespaces" << YAML::Value << f.namespaces;
90 if (!f.access.empty())
91 out << YAML::Key << "access" << YAML::Value << f.access;
92 if (!f.context.empty())
93 out << YAML::Key << "context" << YAML::Value << f.context;
94 if (!f.dependants.empty())
95 out << YAML::Key << "dependants" << YAML::Value << f.dependants;
96 if (!f.dependencies.empty())
97 out << YAML::Key << "dependencies" << YAML::Value << f.dependencies;
98 if (!f.elements.empty())
99 out << YAML::Key << "elements" << YAML::Value << f.elements;
100 if (!f.element_types.empty())
101 out << YAML::Key << "element_types" << YAML::Value << f.element_types;
102 if (!f.method_types.empty())
103 out << YAML::Key << "method_types" << YAML::Value << f.method_types;
104 if (!f.paths.empty())
105 out << YAML::Key << "paths" << YAML::Value << f.paths;
106 if (!f.relationships.empty())
107 out << YAML::Key << "relationships" << YAML::Value << f.relationships;
108 if (!f.specializations.empty())
109 out << YAML::Key << "specializations" << YAML::Value
110 << f.specializations;
111 if (!f.subclasses.empty())
112 out << YAML::Key << "subclasses" << YAML::Value << f.subclasses;
113 if (!f.parents.empty())
114 out << YAML::Key << "parents" << YAML::Value << f.parents;
115
116 out << YAML::EndMap;
117 return out;
118}

225YAML::Emitter &operator<<(YAML::Emitter &out, const config &c)
226{
227 out << YAML::BeginMap;
228
229 out << c.compilation_database_dir;
230 out << c.output_directory;
231 out << c.add_compile_flags;
232 out << c.remove_compile_flags;
233
234 out << dynamic_cast<const inheritable_diagram_options &>(c);
235
236 out << YAML::Key << "diagrams";
237 out << YAML::BeginMap;
238
239 for (const auto &[k, v] : c.diagrams) {
240 out << YAML::Key << k;
241 if (v->type() == common::model::diagram_t::kClass) {
242 out << YAML::Value << dynamic_cast<class_diagram &>(*v);
243 }
244 else if (v->type() == common::model::diagram_t::kSequence) {
245 out << YAML::Value << dynamic_cast<sequence_diagram &>(*v);
246 }
247 else if (v->type() == common::model::diagram_t::kInclude) {
248 out << YAML::Value << dynamic_cast<include_diagram &>(*v);
249 }
250 else if (v->type() == common::model::diagram_t::kPackage) {
251 out << YAML::Value << dynamic_cast<package_diagram &>(*v);
252 }
253 }
254
255 out << YAML::EndMap;
256 out << YAML::EndMap;
257 return out;
258}
