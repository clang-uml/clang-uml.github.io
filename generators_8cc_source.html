24void find_translation_units_for_diagrams(
25 const std::vector<std::string> &diagram_names,
26 clanguml::config::config &config,
27 const std::vector<std::string> &compilation_database_files,
28 std::map<std::string, std::vector<std::string>> &translation_units_map)
29{
30 for (const auto &[name, diagram] : config.diagrams) {
31 // If there are any specific diagram names provided on the command line,
32 // and this diagram is not in that list - skip it
33 if (!diagram_names.empty() && !util::contains(diagram_names, name))
34 continue;
35
36 // If glob is not defined use all translation units from the
37 // compilation database
38 if (!diagram->glob.has_value) {
39 translation_units_map[name] = compilation_database_files;
40 }
41 // Otherwise, get all translation units matching the glob from diagram
42 // configuration
43 else {
44 const std::vector<std::string> translation_units =
45 diagram->get_translation_units();
46
47 std::vector<std::string> valid_translation_units{};
48 std::copy_if(compilation_database_files.begin(),
49 compilation_database_files.end(),
50 std::back_inserter(valid_translation_units),
51 [&translation_units](const auto &tu) {
52 return util::contains(translation_units, tu);
53 });
54
55 translation_units_map[name] = std::move(valid_translation_units);
56 }
57 }
58}

62template <typename DiagramConfig, typename GeneratorTag, typename DiagramModel>
63void generate_diagram_select_generator(const std::string &od,
64 const std::string &name, std::shared_ptr<clanguml::config::diagram> diagram,
65 const DiagramModel &model)
66{
67 using diagram_generator =
68 typename diagram_generator_t<DiagramConfig, GeneratorTag>::type;
69
70 auto path = std::filesystem::path{od} /
71 fmt::format("{}.{}", name, GeneratorTag::extension);
72 std::ofstream ofs;
73 ofs.open(path, std::ofstream::out | std::ofstream::trunc);
74 ofs << diagram_generator(dynamic_cast<DiagramConfig &>(*diagram), *model);
75
76 ofs.close();
77
78 LOG_INFO("Written {} diagram to {}", name, path.string());
79}

81template <typename DiagramConfig>
82void generate_diagram_impl(const std::string &od, const std::string &name,
83 std::shared_ptr<clanguml::config::diagram> diagram,
84 const common::compilation_database &db,
85 const std::vector<std::string> &translation_units,
86 const cli::runtime_config &rc, std::function<void()> &&progress)
87{
88 using diagram_config = DiagramConfig;
89 using diagram_model = typename diagram_model_t<DiagramConfig>::type;
90 using diagram_visitor = typename diagram_visitor_t<DiagramConfig>::type;
91
92 auto model = clanguml::common::generators::generate<diagram_model,
93 diagram_config, diagram_visitor>(db, diagram->name,
94 dynamic_cast<diagram_config &>(*diagram), translation_units, rc.verbose,
95 std::move(progress));
96
97 if constexpr (std::is_same_v<DiagramConfig, config::sequence_diagram>) {
98 if (rc.print_from) {
99 auto from_values = model->list_from_values();
100
101 for (const auto &from : from_values) {
102 std::cout << from << std::endl;
103 }
104
105 return;
106 }
107 if (rc.print_to) {
108 auto to_values = model->list_to_values();
109
110 for (const auto &to : to_values) {
111 std::cout << "|" << to << "|" << std::endl;
112 }
113
114 return;
115 }
116 }
117
118 for (const auto generator_type : rc.generators) {
119 if (generator_type == generator_type_t::plantuml) {
120 generate_diagram_select_generator<diagram_config,
121 plantuml_generator_tag>(od, name, diagram, model);
122 }
123 else if (generator_type == generator_type_t::json) {
124 generate_diagram_select_generator<diagram_config,
125 json_generator_tag>(od, name, diagram, model);
126 }
127 }
128}

131void generate_diagram(const std::string &od, const std::string &name,
132 std::shared_ptr<clanguml::config::diagram> diagram,
133 const common::compilation_database &db,
134 const std::vector<std::string> &translation_units,
135 const cli::runtime_config &runtime_config, std::function<void()> &&progress)
136{
137 using clanguml::common::generator_type_t;
138 using clanguml::common::model::diagram_t;
139
140 using clanguml::config::class_diagram;
141 using clanguml::config::include_diagram;
142 using clanguml::config::package_diagram;
143 using clanguml::config::sequence_diagram;
144
145 if (diagram->type() == diagram_t::kClass) {
146 detail::generate_diagram_impl<class_diagram>(od, name, diagram, db,
147 translation_units, runtime_config, std::move(progress));
148 }
149 else if (diagram->type() == diagram_t::kSequence) {
150 detail::generate_diagram_impl<sequence_diagram>(od, name, diagram, db,
151 translation_units, runtime_config, std::move(progress));
152 }
153 else if (diagram->type() == diagram_t::kPackage) {
154 detail::generate_diagram_impl<package_diagram>(od, name, diagram, db,
155 translation_units, runtime_config, std::move(progress));
156 }
157 else if (diagram->type() == diagram_t::kInclude) {
158 detail::generate_diagram_impl<include_diagram>(od, name, diagram, db,
159 translation_units, runtime_config, std::move(progress));
160 }
161}

163void generate_diagrams(const std::vector<std::string> &diagram_names,
164 config::config &config, const std::string &od,
165 const common::compilation_database_ptr &db,
166 const cli::runtime_config &runtime_config,
167 const std::map<std::string, std::vector<std::string>>
168 &translation_units_map)
169{
170 util::thread_pool_executor generator_executor{runtime_config.thread_count};
171 std::vector<std::future<void>> futs;
172
173 std::unique_ptr<progress_indicator> indicator;
174
175 if (runtime_config.progress) {
176 std::cout << termcolor::white
177 << "Processing translation units and generating diagrams:\n";
178 indicator = std::make_unique<progress_indicator>();
179 }
180
181 for (const auto &[name, diagram] : config.diagrams) {
182 // If there are any specific diagram names provided on the command
183 // line, and this diagram is not in that list - skip it
184 if (!diagram_names.empty() && !util::contains(diagram_names, name))
185 continue;
186
187 const auto &valid_translation_units = translation_units_map.at(name);
188
189 if (valid_translation_units.empty()) {
190 if (indicator) {
191 indicator->add_progress_bar(
192 name, 0, diagram_type_to_color(diagram->type()));
193 indicator->fail(name);
194 }
195 else {
196 LOG_ERROR(
197 "Diagram {} generation failed: no translation units "
198 "found. Please make sure that your 'glob' patterns match "
199 "at least 1 file in 'compile_commands.json'.",
200 name);
201 }
202 continue;
203 }
204
205 auto generator = [&od, &name = name, &diagram = diagram, &indicator,
206 db = std::ref(*db),
207 translation_units = valid_translation_units,
208 runtime_config]() mutable {
209 try {
210 if (indicator)
211 indicator->add_progress_bar(name, translation_units.size(),
212 diagram_type_to_color(diagram->type()));
213
214 generate_diagram(od, name, diagram, db, translation_units,
215 runtime_config, [&indicator, &name]() {
216 if (indicator)
217 indicator->increment(name);
218 });
219
220 if (indicator)
221 indicator->complete(name);
222 }
223 catch (std::exception &e) {
224 if (indicator)
225 indicator->fail(name);
226
227 LOG_ERROR(e.what());
228 }
229 };
230
231 futs.emplace_back(generator_executor.add(std::move(generator)));
232 }
233
234 for (auto &fut : futs) {
235 fut.get();
236 }
237
238 if (runtime_config.progress) {
239 indicator->stop();
240 std::cout << termcolor::white << "Done\n";
241 std::cout << termcolor::reset;
242 }
243}

245indicators::Color diagram_type_to_color(model::diagram_t diagram_type)
246{
247 switch (diagram_type) {
248 case model::diagram_t::kClass:
249 return indicators::Color::yellow;
250 case model::diagram_t::kSequence:
251 return indicators::Color::blue;
252 case model::diagram_t::kPackage:
253 return indicators::Color::cyan;
254 case model::diagram_t::kInclude:
255 return indicators::Color::magenta;
256 default:
257 return indicators::Color::unspecified;
258 }
