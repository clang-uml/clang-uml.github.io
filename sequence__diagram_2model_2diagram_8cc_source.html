101void diagram::add_message(model::message &&message)
102{
103 const auto caller_id = message.from();
104 if (sequences_.find(caller_id) == sequences_.end()) {
105 activity a{caller_id};
106 sequences_.insert({caller_id, std::move(a)});
107 }
108
109 get_activity(caller_id).add_message(std::move(message));
110}

184void diagram::print() const
185{
186 LOG_TRACE(" --- Participants ---");
187 for (const auto &[id, participant] : participants_) {
188 LOG_DBG("{} - {}", id, participant->to_string());
189 }
190
191 LOG_TRACE(" --- Activities ---");
192 for (const auto &[from_id, act] : sequences_) {
193
194 LOG_TRACE("Sequence id={}:", from_id);
195
196 const auto &from_activity = *(participants_.at(from_id));
197
198 LOG_TRACE(" Activity id={}, from={}:", act.from(),
199 from_activity.full_name(false));
200
201 for (const auto &message : act.messages()) {
202 if (participants_.find(message.from()) == participants_.end())
203 continue;
204
205 const auto &from_participant = *participants_.at(message.from());
206
207 if (participants_.find(message.to()) == participants_.end()) {
208 LOG_TRACE(" Message from={}, from_id={}, "
209 "to={}, to_id={}, name={}, type={}",
210 from_participant.full_name(false), from_participant.id(),
211 "__UNRESOLVABLE_ID__", message.to(), message.message_name(),
212 to_string(message.type()));
213 }
214 else {
215 const auto &to_participant = *participants_.at(message.to());
216
217 LOG_TRACE(" Message from={}, from_id={}, "
218 "to={}, to_id={}, name={}, type={}",
219 from_participant.full_name(false), from_participant.id(),
220 to_participant.full_name(false), to_participant.id(),
221 message.message_name(), to_string(message.type()));
222 }
223 }
224 }

227void diagram::fold_or_end_block_statement(message &&m,
228 const common::model::message_t statement_begin,
229 std::vector<message> &current_messages) const
230{
231 bool is_empty_statement{true};
232
233 auto rit = current_messages.rbegin();
234 for (; rit != current_messages.rend(); rit++) {
235 if (rit->type() == statement_begin) {
236 break;
237 }
238 if (rit->type() == common::model::message_t::kCall) {
239 is_empty_statement = false;
240 break;
241 }
242 }
243
244 if (is_empty_statement) {
245 current_messages.erase((rit + 1).base(), current_messages.end());
246 }
247 else {
248 current_messages.emplace_back(std::move(m));
249 }
250}

252void diagram::finalize()
253{
254 // Apply diagram filters and remove any empty block statements
255 using common::model::message_t;
256
257 // First in each sequence (activity) filter out any remaining
258 // uninteresting calls
259 for (auto &[id, act] : sequences_) {
260 util::erase_if(act.messages(), [this](auto &m) {
261 if (m.type() != message_t::kCall)
262 return false;
263
264 const auto &to = get_participant<model::participant>(m.to());
265 if (!to || to.value().skip())
266 return true;
267
268 if (!should_include(to.value())) {
269 LOG_DBG("Excluding call from [{}] to {} [{}]", m.from(),
270 to.value().full_name(false), m.to());
271 return true;
272 }
273
274 return false;
275 });
276 }
277
278 // Now remove any empty block statements, e.g. if/endif
279 for (auto &[id, act] : sequences_) {
280 int64_t block_nest_level{0};
281 std::vector<std::vector<message>> block_message_stack;
282 // Add first stack level - this level will contain the filtered
283 // message sequence
284 block_message_stack.emplace_back();
285
286 // First create a recursive stack from the messages and
287 // message blocks (e.g. if statements)
288 for (auto &m : act.messages()) {
289 if (is_begin_block_message(m.type())) {
290 block_nest_level++;
291 block_message_stack.push_back({m});
292 }
293 else if (is_end_block_message(m.type())) {
294 block_nest_level--;
295
296 block_message_stack.back().push_back(m);
297
298 // Check the last stack for any calls, if yes, collapse it
299 // on the previous stack
300 if (std::count_if(block_message_stack.back().begin(),
301 block_message_stack.back().end(), [](auto &m) {
302 return m.type() == message_t::kCall;
303 }) > 0) {
304 std::copy(block_message_stack.back().begin(),
305 block_message_stack.back().end(),
306 std::back_inserter(
307 block_message_stack.at(block_nest_level)));
308 }
309
310 block_message_stack.pop_back();
311
312 assert(block_nest_level >= 0);
313 }
314 else {
315 if (m.type() == message_t::kCall) {
316 // Set the message return type based on the callee return
317 // type
318 auto to_participant =
319 get_participant<sequence_diagram::model::function>(
320 m.to());
321 if (to_participant.has_value()) {
322 m.set_return_type(to_participant.value().return_type());
323 }
324 }
325 block_message_stack.back().push_back(m);
326 }
327 }
328
329 act.messages().clear();
330
331 for (auto &m : block_message_stack[0]) {
332 act.add_message(m);
333 }
334 }
335}
