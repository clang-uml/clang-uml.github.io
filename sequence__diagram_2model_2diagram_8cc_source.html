76void diagram::add_participant(std::unique_ptr<participant> p)

196std::vector<std::string> diagram::list_from_values() const
197{
198 std::vector<std::string> result;
199
200 for (const auto &[from_id, act] : sequences_) {
201
202 const auto &from_activity = *(participants_.at(from_id));
203 const auto &full_name = from_activity.full_name(false);
204 if (!full_name.empty())
205 result.push_back(full_name);
206 }
207
208 std::sort(result.begin(), result.end());
209 result.erase(std::unique(result.begin(), result.end()), result.end());
210
211 return result;

235common::model::diagram_element::id_t diagram::get_to_activity_id(
236 const config::source_location &to_location) const
237{
238 common::model::diagram_element::id_t to_activity{0};
239
240 for (const auto &[k, v] : sequences()) {
241 for (const auto &m : v.messages()) {
242 if (m.type() != common::model::message_t::kCall)
243 continue;
244 const auto &callee = *participants().at(m.to());
245 std::string vto = callee.full_name(false);
246 if (vto == to_location.location) {
247 LOG_DBG(
248 "Found sequence diagram end point '{}': {}", vto, m.to());
249 to_activity = m.to();
250 break;
251 }
252 }
253 }
254
255 if (to_activity == 0) {
256 LOG_WARN("Failed to find 'to' participant {} for to "
257 "condition",
258 to_location.location);
259 }
260
261 return to_activity;
262}

264common::model::diagram_element::id_t diagram::get_from_activity_id(
265 const config::source_location &from_location) const
266{
267 common::model::diagram_element::id_t from_activity{0};
268
269 for (const auto &[k, v] : sequences()) {
270 const auto &caller = *participants().at(v.from());
271 std::string vfrom = caller.full_name(false);
272 if (vfrom == from_location.location) {
273 LOG_DBG("Found sequence diagram start point '{}': {}", vfrom, k);
274 from_activity = k;
275 break;
276 }
277 }
278
279 if (from_activity == 0) {
280 LOG_WARN("Failed to find 'from' participant {} for from "
281 "condition",
282 from_location.location);
283 }
284
285 return from_activity;

288std::vector<message_chain_t> diagram::get_all_from_to_message_chains(
289 const common::model::diagram_element::id_t from_activity,
290 const common::model::diagram_element::id_t to_activity) const
291{
292 std::vector<message_chain_t> message_chains_unique{};
293
294 // Message (call) chains matching the specified from_to condition
295 std::vector<message_chain_t> message_chains;
296
297 // First find all 'to_activity' call targets in the sequences, i.e.
298 // all messages pointing to the final 'to_activity' activity
299 for (const auto &[k, v] : sequences()) {
300 for (const auto &m : v.messages()) {
301 if (m.type() != common::model::message_t::kCall)
302 continue;
303
304 if (m.to() == to_activity) {
305 message_chains.emplace_back();
306 message_chains.back().push_back(m);
307 }
308 }
309 }
310
311 std::map<unsigned int, std::vector<model::message>> calls_to_current_chain;
312 std::map<unsigned int, message_chain_t> current_chain;
313
314 int iter = 0;
315 while (true) {
316 bool added_message_to_some_chain{false};
317 // If target of current message matches any of the
318 // 'from' constraints in the last messages in
319 // current chains found on previous iteration - append
320 if (!calls_to_current_chain.empty()) {
321 for (auto &[message_chain_index, messages] :
322 calls_to_current_chain) {
323 for (auto &m : messages) {
324 message_chains.push_back(
325 current_chain[message_chain_index]);
326
327 message_chains.back().push_back(std::move(m));
328 }
329 }
330 calls_to_current_chain.clear();
331 }
332
333 LOG_TRACE("Message chains after iteration {}", iter++);
334 int message_chain_index{};
335 for (const auto &mc : message_chains) {
336 LOG_TRACE("\t{}: {}", message_chain_index++,
337 fmt::join(util::map<std::string>(mc,
338 [](const model::message &m) -> std::string {
339 return m.message_name();
340 }),
341 "<-"));
342 }
343
344 for (auto i = 0U; i < message_chains.size(); i++) {
345 auto &mc = message_chains[i];
346 current_chain[i] = mc;
347 for (const auto &[k, v] : sequences()) {
348 for (const auto &m : v.messages()) {
349 if (m.type() != common::model::message_t::kCall)
350 continue;
351
352 // Ignore recursive calls and call loops
353 if (m.to() == m.from() ||
354 std::any_of(
355 cbegin(mc), cend(mc), [&m](const auto &msg) {
356 return msg.to() == m.from();
357 })) {
358 continue;
359 }
360
361 if (m.to() == mc.back().from()) {
362 calls_to_current_chain[i].push_back(m);
363 added_message_to_some_chain = true;
364 }
365 }
366 }
367
368 // If there are more than one call to the current chain,
369 // duplicate it as many times as there are calls - 1
370 if (calls_to_current_chain.count(i) > 0 &&
371 !calls_to_current_chain[i].empty()) {
372 mc.push_back(calls_to_current_chain[i][0]);
373 calls_to_current_chain[i].erase(
374 calls_to_current_chain[i].begin());
375 }
376 }
377
378 // There is nothing more to find
379 if (!added_message_to_some_chain)
380 break;
381 }
382
383 // Reverse the message chains order (they were added starting from
384 // the destination activity)
385 for (auto &mc : message_chains) {
386 std::reverse(mc.begin(), mc.end());
387
388 if (mc.empty())
389 continue;
390
391 if (std::find(message_chains_unique.begin(),
392 message_chains_unique.end(), mc) != message_chains_unique.end())
393 continue;
394
395 if (from_activity == 0 || (mc.front().from() == from_activity)) {
396 message_chains_unique.push_back(mc);
397 }
398 }
399
400 LOG_TRACE("Message chains unique", iter++);
401 int message_chain_index{};
402 for (const auto &mc : message_chains_unique) {
403 LOG_TRACE("\t{}: {}", message_chain_index++,
404 fmt::join(util::map<std::string>(mc,
405 [](const model::message &m) -> std::string {
406 return m.message_name();
407 }),
408 "->"));
409 }
410
411 return message_chains_unique;
412}

414void diagram::print() const
415{
416 LOG_TRACE(" --- Participants ---");
417 for (const auto &[id, participant] : participants_) {
418 LOG_DBG("{} - {}", id, participant->to_string());
419 }
420
421 LOG_TRACE(" --- Activities ---");
422 for (const auto &[from_id, act] : sequences_) {
423
424 LOG_TRACE("Sequence id={}:", from_id);
425
426 const auto &from_activity = *(participants_.at(from_id));
427
428 LOG_TRACE(" Activity id={}, from={}:", act.from(),
429 from_activity.full_name(false));
430
431 for (const auto &message : act.messages()) {
432 if (participants_.find(message.from()) == participants_.end())
433 continue;
434
435 const auto &from_participant = *participants_.at(message.from());
436
437 if (participants_.find(message.to()) == participants_.end()) {
438 LOG_TRACE(" Message from={}, from_id={}, "
439 "to={}, to_id={}, name={}, type={}",
440 from_participant.full_name(false), from_participant.id(),
441 "__UNRESOLVABLE_ID__", message.to(), message.message_name(),
442 to_string(message.type()));
443 }
444 else {
445 const auto &to_participant = *participants_.at(message.to());
446
447 LOG_TRACE(" Message from={}, from_id={}, "
448 "to={}, to_id={}, name={}, type={}, comment={}",
449 from_participant.full_name(false), from_participant.id(),
450 to_participant.full_name(false), to_participant.id(),
451 message.message_name(), to_string(message.type()),
452 message.comment() ? message.comment().value() : "None");
453 }
454 }
455 }

458void diagram::fold_or_end_block_statement(message &&m,
459 const common::model::message_t statement_begin,
460 std::vector<message> &current_messages) const
461{
462 bool is_empty_statement{true};
463
464 auto rit = current_messages.rbegin();
465 for (; rit != current_messages.rend(); rit++) {
466 if (rit->type() == statement_begin) {
467 break;
468 }
469 if (rit->type() == common::model::message_t::kCall) {
470 is_empty_statement = false;
471 break;
472 }
473 }
474
475 if (is_empty_statement) {
476 current_messages.erase((rit + 1).base(), current_messages.end());
477 }
478 else {
479 current_messages.emplace_back(std::move(m));
480 }
481}

483void diagram::finalize()
484{
485 // Apply diagram filters and remove any empty block statements
486 using common::model::message_t;
487
488 // First in each sequence (activity) filter out any remaining
489 // uninteresting calls
490 for (auto &[id, act] : sequences_) {
491 util::erase_if(act.messages(), [this](auto &m) {
492 if (m.type() != message_t::kCall)
493 return false;
494
495 const auto &to = get_participant<model::participant>(m.to());
496 if (!to || to.value().skip())
497 return true;
498
499 if (!should_include(to.value())) {
500 LOG_DBG("Excluding call from [{}] to {} [{}]", m.from(),
501 to.value().full_name(false), m.to());
502 return true;
503 }
504
505 return false;
506 });
507 }
508
509 // Now remove any empty block statements, e.g. if/endif
510 for (auto &[id, act] : sequences_) {
511 int64_t block_nest_level{0};
512 std::vector<std::vector<message>> block_message_stack;
513 // Add first stack level - this level will contain the filtered
514 // message sequence
515 block_message_stack.emplace_back();
516
517 // First create a recursive stack from the messages and
518 // message blocks (e.g. if statements)
519 for (auto &m : act.messages()) {
520 if (is_begin_block_message(m.type())) {
521 block_nest_level++;
522 block_message_stack.push_back({m});
523 }
524 else if (is_end_block_message(m.type())) {
525 block_nest_level--;
526
527 block_message_stack.back().push_back(m);
528
529 // Check the last stack for any calls, if yes, collapse it
530 // on the previous stack
531 if (std::count_if(block_message_stack.back().begin(),
532 block_message_stack.back().end(), [](auto &m) {
533 return m.type() == message_t::kCall;
534 }) > 0) {
535 std::copy(block_message_stack.back().begin(),
536 block_message_stack.back().end(),
537 std::back_inserter(
538 block_message_stack.at(block_nest_level)));
539 }
540
541 block_message_stack.pop_back();
542
543 assert(block_nest_level >= 0);
544 }
545 else {
546 if (m.type() == message_t::kCall) {
547 // Set the message return type based on the callee return
548 // type
549 auto to_participant =
550 get_participant<sequence_diagram::model::function>(
551 m.to());
552 if (to_participant.has_value()) {
553 m.set_return_type(to_participant.value().return_type());
554 }
555 }
556 block_message_stack.back().push_back(m);
557 }
558 }
559
560 act.messages().clear();
561
562 for (auto &m : block_message_stack[0]) {
563 act.add_message(m);
564 }
565 }
