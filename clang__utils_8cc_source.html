25model::access_t access_specifier_to_access_t(
26 clang::AccessSpecifier access_specifier)
27{
28 auto access = model::access_t::kPublic;
29 switch (access_specifier) {
30 case clang::AccessSpecifier::AS_public:
31 access = model::access_t::kPublic;
32 break;
33 case clang::AccessSpecifier::AS_private:
34 access = model::access_t::kPrivate;
35 break;
36 case clang::AccessSpecifier::AS_protected:
37 access = model::access_t::kProtected;
38 break;
39 default:
40 break;
41 }
42
43 return access;
44}

46model::namespace_ get_tag_namespace(const clang::TagDecl &declaration)
47{
48 model::namespace_ ns;
49
50 const auto *parent{declaration.getParent()};
51
52 // First walk up to the nearest namespace, e.g. from nested class or enum
53 while ((parent != nullptr) && !parent->isNamespace()) {
54 parent = parent->getParent();
55 }
56
57 // Now build up the namespace
58 std::deque<std::string> namespace_tokens;
59 while ((parent != nullptr) && parent->isNamespace()) {
60 if (const auto *ns_decl = clang::dyn_cast<clang::NamespaceDecl>(parent);
61 ns_decl != nullptr) {
62 if (!ns_decl->isInline() && !ns_decl->isAnonymousNamespace())
63 namespace_tokens.push_front(ns_decl->getNameAsString());
64 }
65
66 parent = parent->getParent();
67 }
68
69 for (const auto &ns_token : namespace_tokens) {
70 ns |= ns_token;
71 }
72
73 return ns;
74}

84std::string get_tag_name(const clang::TagDecl &declaration)
85{
86 auto base_name = declaration.getNameAsString();
87
88 if (base_name.empty()) {
89 base_name =
90 fmt::format("(anonymous_{})", std::to_string(declaration.getID()));
91 }
92
93 if ((declaration.getParent() != nullptr) &&
94 declaration.getParent()->isRecord()) {
95 // If the record is nested within another record (e.g. class or struct)
96 // we have to maintain a containment namespace in order to ensure
97 // unique names within the diagram
98 std::deque<std::string> record_parent_names;
99 record_parent_names.push_front(base_name);
100
101 const auto *cls_parent{declaration.getParent()};
102 while (cls_parent->isRecord()) {
103 if (const auto *record_decl =
104 clang::dyn_cast<clang::RecordDecl>(cls_parent);
105 record_decl != nullptr) {
106 record_parent_names.push_front(record_decl->getNameAsString());
107 }
108 cls_parent = cls_parent->getParent();
109 }
110 return fmt::format("{}", fmt::join(record_parent_names, "##"));
111 }
112
113 return base_name;
114}

116std::string to_string(const clang::QualType &type, const clang::ASTContext &ctx,
117 bool try_canonical)
118{
119 clang::PrintingPolicy print_policy(ctx.getLangOpts());
120 print_policy.SuppressScope = 0;
121 print_policy.PrintCanonicalTypes = 0;
122
123 std::string result;
124
125 result = type.getAsString(print_policy);
126
127 if (try_canonical && result.find('<') != std::string::npos) {
128 auto canonical_type_name =
129 type.getCanonicalType().getAsString(print_policy);
130
131 auto result_qualified_template_name =
132 result.substr(0, result.find('<'));
133 auto result_template_arguments = result.substr(result.find('<'));
134
135 auto canonical_qualified_template_name =
136 canonical_type_name.substr(0, canonical_type_name.find('<'));
137
138 // Choose the longer name (why do I have to do this?)
139 if (result_qualified_template_name.size() <
140 canonical_qualified_template_name.size()) {
141
142 result =
143 canonical_qualified_template_name + result_template_arguments;
144 }
145 }
146
147 // If for any reason clang reports the type as empty string, make sure
148 // it has some default name
149 if (result.empty())
150 result = "(anonymous)";
151 else if (util::contains(result, "unnamed struct") ||
152 util::contains(result, "unnamed union")) {
153 result = common::get_tag_name(*type->getAsTagDecl());
154 }
155 else if (util::contains(result, "anonymous struct") ||
156 util::contains(result, "anonymous union")) {
157 result = common::get_tag_name(*type->getAsTagDecl());
158 }
159
160 // Remove trailing spaces after commas in template arguments
161 clanguml::util::replace_all(result, ", ", ",");
162 clanguml::util::replace_all(result, "> >", ">>");
163
164 // Try to get rid of 'type-parameter-X-Y' ugliness
165 if (result.find("type-parameter-") != std::string::npos) {
166 util::if_not_null(
167 common::dereference(type)->getAs<clang::TypedefType>(),
168 [&result, &type](auto *p) {
169 auto [unqualified_type, context] =
170 common::consume_type_context(type);
171 result = p->getDecl()->getNameAsString();
172 if (!context.empty()) {
173 std::vector<std::string> deduced_contexts;
174
175 for (const auto &c : context) {
176 deduced_contexts.push_back(c.to_string());
177 }
178
179 result = fmt::format(
180 "{} {}", result, fmt::join(deduced_contexts, " "));
181 }
182 });
183 }
184
185 return result;
186}

194std::string to_string(
195 const clang::TemplateArgument &arg, const clang::ASTContext *ctx)
196{
197 switch (arg.getKind()) {
198 case clang::TemplateArgument::Expression:
199 return to_string(arg.getAsExpr());
200 case clang::TemplateArgument::Type:
201 return to_string(arg.getAsType(), *ctx, false);
202 case clang::TemplateArgument::Null:
203 return "";
204 case clang::TemplateArgument::NullPtr:
205 return "nullptr";
206 case clang::TemplateArgument::Integral:
207 return std::to_string(arg.getAsIntegral().getExtValue());
208 case clang::TemplateArgument::Template:
209 return to_string(arg.getAsTemplate());
210 case clang::TemplateArgument::TemplateExpansion:
211 return to_string(arg.getAsTemplateOrTemplatePattern());
212 default:
213 return "";
214 }
215}

302{
303 const clang::LangOptions lo;
304

309 return get_source_text_raw(printable_range, sm);

312std::tuple<unsigned int, unsigned int, std::string>
313extract_template_parameter_index(const std::string &type_parameter)
314{
315 assert(type_parameter.find("type-parameter-") == 0);
316
317 auto type_parameter_and_suffix = util::split(type_parameter, " ");
318
319 auto toks = util::split(
320 type_parameter_and_suffix.front().substr(strlen("type-parameter-")),
321 "-");
322
323 std::string qualifier;
324
325 if (type_parameter_and_suffix.size() > 1) {
326 qualifier = type_parameter_and_suffix.at(1);
327 }
328
329 return {std::stoi(toks.at(0)), std::stoi(toks.at(1)), std::move(qualifier)};
330}

332bool is_subexpr_of(const clang::Stmt *parent_stmt, const clang::Stmt *sub_stmt)
333{
334 if (parent_stmt == nullptr || sub_stmt == nullptr)
335 return false;
336
337 if (parent_stmt == sub_stmt)
338 return true;
339
340 return std::any_of(parent_stmt->child_begin(), parent_stmt->child_end(),
341 [sub_stmt](const auto *e) { return is_subexpr_of(e, sub_stmt); });

416std::vector<common::model::template_parameter> parse_unexposed_template_params(
417 const std::string &params,
418 const std::function<std::string(const std::string &)> &ns_resolve,
419 int depth)
420{
421 using common::model::template_parameter;
422
423 std::vector<template_parameter> res;
424
425 auto it = params.begin();
426 while (std::isspace(*it) != 0)
427 ++it;
428
429 std::string type{};
430 std::vector<template_parameter> nested_params;
431 bool complete_class_template_argument{false};
432
433 while (it != params.end()) {
434 if (*it == '<') {
435 int nested_level{0};
436 auto bracket_match_begin = it + 1;
437 auto bracket_match_end = bracket_match_begin;
438 while (bracket_match_end != params.end()) {
439 if (*bracket_match_end == '<') {
440 nested_level++;
441 }
442 else if (*bracket_match_end == '>') {
443 if (nested_level > 0)
444 nested_level--;
445 else
446 break;
447 }
448 else {
449 }
450 bracket_match_end++;
451 }
452
453 std::string nested_params_str(
454 bracket_match_begin, bracket_match_end);
455
456 nested_params = parse_unexposed_template_params(
457 nested_params_str, ns_resolve, depth + 1);
458
459 if (nested_params.empty()) {
460 // We couldn't extract any nested template parameters from
461 // `nested_params_str` so just add it as type of template
462 // argument as is
463 nested_params.emplace_back(
464 template_parameter::make_unexposed_argument(
465 nested_params_str));
466 }
467
468 it = bracket_match_end - 1;
469 }
470 else if (*it == '>') {
471 complete_class_template_argument = true;
472 if (depth == 0) {
473 break;
474 }
475 }
476 else if (*it == ',') {
477 complete_class_template_argument = true;
478 }
479 else {
480 type += *it;
481 }
482 if (complete_class_template_argument) {
483 auto t = template_parameter::make_unexposed_argument(
484 ns_resolve(clanguml::util::trim_typename(type)));
485 type = "";
486 for (auto &&param : nested_params)
487 t.add_template_param(std::move(param));
488
489 res.emplace_back(std::move(t));
490 complete_class_template_argument = false;
491 }
492 it++;
493 }
494
495 if (!type.empty()) {
496 auto t = template_parameter::make_unexposed_argument(
497 ns_resolve(clanguml::util::trim_typename(type)));
498 type = "";
499 for (auto &&param : nested_params)
500 t.add_template_param(std::move(param));
501
502 res.emplace_back(std::move(t));
503 }
504
505 return res;
506}

567std::string format_condition_text(const std::string &condition_text)
568{
569 std::string result{condition_text};
570
571 if (result.size() < 2)
572 return {};
573
574 std::vector<std::string> text_lines = util::split(result, "\n", true);
575
576 // Trim each line
577 for (auto &line : text_lines) {
578 line = util::trim(line);
579 }
580
581 result = util::join(" ", text_lines);
582
583 if (result.at(0) == '(' && result.back() == ')')
584 return result.substr(1, result.size() - 2);
585
586 return result;
587}

636clang::QualType dereference(clang::QualType type)
637{
638 auto res = type;
639
640 while (true) {
641 if (res->isReferenceType())
642 res = res.getNonReferenceType();
643 else if (res->isPointerType())
644 res = res->getPointeeType();
645 else
646 break;
647 }
648
649 return res;
650}

652std::pair<clang::QualType, std::deque<common::model::context>>
653consume_type_context(clang::QualType type)
654{
655 std::deque<common::model::context> res;
656
657 while (true) {
658 bool try_again{false};
659 common::model::context ctx;
660
661 if (type.isConstQualified()) {
662 ctx.is_const = true;
663 try_again = true;
664 }
665
666 if (type.isVolatileQualified()) {
667 ctx.is_volatile = true;
668 try_again = true;
669 }
670
671 if (type->isPointerType() || type->isReferenceType()) {
672 if (type.isConstQualified() || type.isVolatileQualified()) {
673 ctx.is_ref_const = type.isConstQualified();
674 ctx.is_ref_volatile = type.isVolatileQualified();
675
676 try_again = true;
677 }
678 }
679
680 if (type->isLValueReferenceType()) {
681 ctx.pr = common::model::rpqualifier::kLValueReference;
682 try_again = true;
683 }
684 else if (type->isRValueReferenceType()) {
685 ctx.pr = common::model::rpqualifier::kRValueReference;
686 try_again = true;
687 }
688 else if (type->isMemberFunctionPointerType() &&
689 type->getPointeeType()->getAs<clang::FunctionProtoType>() !=
690 nullptr) {
691 const auto ref_qualifier =
692 type->getPointeeType() // NOLINT
693 ->getAs<clang::FunctionProtoType>() // NOLINT
694 ->getRefQualifier();
695
696 if (ref_qualifier == clang::RefQualifierKind::RQ_RValue) {
697 ctx.pr = common::model::rpqualifier::kRValueReference;
698 try_again = true;
699 }
700 else if (ref_qualifier == clang::RefQualifierKind::RQ_LValue) {
701 ctx.pr = common::model::rpqualifier::kLValueReference;
702 try_again = true;
703 }
704 }
705 else if (type->isPointerType()) {
706 ctx.pr = common::model::rpqualifier::kPointer;
707 try_again = true;
708 }
709
710 if (try_again) {
711 if (type->isPointerType()) {
712 if (type->getPointeeType().isConstQualified())
713 ctx.is_const = true;
714 if (type->getPointeeType().isVolatileQualified())
715 ctx.is_volatile = true;
716
717 type = type->getPointeeType().getUnqualifiedType();
718 }
719 else if (type->isReferenceType()) {
720 if (type.getNonReferenceType().isConstQualified())
721 ctx.is_const = true;
722 if (type.getNonReferenceType().isVolatileQualified())
723 ctx.is_volatile = true;
724
725 type = type.getNonReferenceType().getUnqualifiedType();
726 }
727 else if (type.isConstQualified() || type.isVolatileQualified()) {
728 ctx.is_const = type.isConstQualified();
729 ctx.is_volatile = type.isVolatileQualified();
730 type = type.getUnqualifiedType();
731 }
732
733 res.push_front(ctx);
734
735 if (type->isMemberFunctionPointerType())
736 return std::make_pair(type, res);
737 }
738 else
739 return std::make_pair(type, res);
740 }
741}

743std::vector<std::string> tokenize_unexposed_template_parameter(
744 const std::string &t)
745{
746 std::vector<std::string> result;
747
748 auto spaced_out = util::split(t, " ");
749
750 for (const auto &word : spaced_out) {
751 if (is_qualified_identifier(word)) {
752 if (word != "class" && word != "templated" && word != "struct")
753 result.emplace_back(word);
754 continue;
755 }
756
757 std::string tok;
758
759 for (const char c : word) {
760 if (c == '(' || c == ')' || c == '[' || c == ']' || c == '<' ||
761 c == '>') {
762 if (!tok.empty())
763 result.emplace_back(tok);
764 result.emplace_back(std::string{c});
765 tok.clear();
766 }
767 else if (c == ':') {
768 if (!tok.empty() && tok != ":") {
769 result.emplace_back(tok);
770 tok = ":";
771 }
772 else if (tok == ":") {
773 result.emplace_back("::");
774 tok = "";
775 }
776 else {
777 tok += ':';
778 }
779 }
780 else if (c == ',') {
781 if (!tok.empty()) {
782 result.emplace_back(tok);
783 }
784 result.emplace_back(",");
785 tok.clear();
786 }
787 else if (c == '*') {
788 if (!tok.empty()) {
789 result.emplace_back(tok);
790 }
791 result.emplace_back("*");
792 tok.clear();
793 }
794 else if (c == '.') {
795 // This can only be the case if we have a variadic template,
796 // right?
797 if (tok == "..") {
798 result.emplace_back("...");
799 tok.clear();
800 }
801 else if (tok == ".") {
802 tok = "..";
803 }
804 else if (!tok.empty()) {
805 result.emplace_back(tok);
806 tok = ".";
807 }
808 }
809 else {
810 tok += c;
811 }
812 }
813
814 tok = util::trim(tok);
815
816 if (!tok.empty()) {
817 if (tok != "class" && tok != "typename" && word != "struct")
818 result.emplace_back(tok);
819 tok.clear();
820 }
821 }
822
823 return result;
824}

826bool parse_source_location(const std::string &location_str, std::string &file,
827 unsigned &line, unsigned &column)
828{
829 auto tokens = util::split(location_str, ":");
830
831 if (tokens.size() < 3)
832 return false;
833
834 file = tokens.at(0);
835 try {
836 line = std::stoi(tokens.at(1));
837 }
838 catch (std::invalid_argument &e) {
839 line = 0;
840 }
841
842 try {
843 column = std::stoi(tokens.at(2));
844 }
845 catch (std::invalid_argument &e) {
846 column = 0;
847 }
848
849 return true;
850}

852clang::RawComment *get_expression_raw_comment(const clang::SourceManager &sm,
853 const clang::ASTContext &context, const clang::Stmt *stmt)
854{
855 // First get the first line of the expression
856 auto expr_begin = stmt->getSourceRange().getBegin();
857 const auto expr_begin_line = sm.getSpellingLineNumber(expr_begin);
858
859 if (!context.Comments.empty() &&
860 context.Comments.getCommentsInFile(sm.getFileID(expr_begin)) != nullptr)
861 for (const auto [offset, raw_comment] :
862 *context.Comments.getCommentsInFile(sm.getFileID(expr_begin))) {
863 const auto comment_end_line = sm.getSpellingLineNumber(
864 raw_comment->getSourceRange().getEnd());
865
866 if (expr_begin_line == comment_end_line ||
867 expr_begin_line == comment_end_line + 1)
868 return raw_comment;
869 }
870
871 return {};
872}
