23std::string to_string(const property_type t)
24{
25 switch (t) {
26 case property_type::kBoolean:
27 return "boolean";
28 case property_type::kInt:
29 return "int";
30 case property_type::kLong:
31 return "long";
32 case property_type::kFloat:
33 return "float";
34 case property_type::kDouble:
35 return "double";
36 case property_type::kString:
37 return "string";
38 default:
39 assert(false);
40 return "";
41 }
42}
